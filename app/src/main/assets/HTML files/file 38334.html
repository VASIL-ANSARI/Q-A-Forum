openerp.my_cool_company = function(instance) {
    instance.web.WebClient.include({
        set_title: function(title) {
            title = _.str.clean(title);
            var sep = _.isEmpty(title) ? '' : ' - ';
            document.title = title + sep + 'My Cool Company';
        },
    });
};
