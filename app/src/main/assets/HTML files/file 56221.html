You can't use Input.GetAxis with the arrow keys but there is a trick on how to do this. First, use Input.GetAxis to get the mouse move value and store it to variable x. If the x value is more or less than 0 then move the Object with the value from Input.GetAxis.

If the x value is 0, check if the left or right arrow is pressed and then make the direction of x to be Vector3.right*speed or Vector3.left.x * keySpeed depending on which key was pressed. Another thing to know is that both speed multipliers must be different values. There should be a speed multiplier for the mouse and another one for the keyboard or the Object will disappear from the screen. The appropriate value for the mouse is 100f and for keyboard is 0.1f. 

public class Paddle : MonoBehaviour
{
    public float mouseSpeed = 100;
    public float keySpeed = 0.1f;


    void Update()
    {
        float x = 0;
        x = Input.GetAxis("Mouse X") * Time.deltaTime * mouseSpeed;

        if (x == 0)
        {
            if (Input.GetKey(KeyCode.LeftArrow))
            {
                x = Vector3.left.x * keySpeed;
            }

            if (Input.GetKey(KeyCode.RightArrow))
            {
                x = Vector3.right.x * keySpeed;
            }

        }
        transform.Translate(x, 0, 0);
    }
}
