Private members can't be accesed by non-member functions. You can make operator<< friend:

class Liste{
    friend ostream& operator<<(ostream& Stream, const Liste &list);
    ...
};

Or add a member function for it:

class Liste{
    public:
    void print(ostream& Stream) const {
        Stream << '[';
        for (int i=0; i < list.anzahl; i++) {
            Stream << list.zahlen[i];
            if (i > (list.anzahl-2) {
                Stream << ',';
            }
        }
        Stream << ']' << endl;
    }
    ...
};

then call it from operator<<:

ostream& operator<<(ostream& Stream, const Liste &list)
{
    list.print(Stream);
    return Stream;
}

BTW: You should use Stream in operator<<, not cout.
