You're not assigning the replacement expression to a variable (.replace is a prototype function and does not assign back to the original property).

Try: 

function myFunction() {
  var textstfu = "<b> I love </b> pancakes <i> I love pancakes </i>";
  textstfu = textstfu.replace(/(<(\D)>)([^<>])+(<\/(\D)>)/gi, "$3");
  Logger.log(textstfu);
}
