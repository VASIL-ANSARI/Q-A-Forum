DECLARE @bCond BIT
SET @bCond = 1

SELECT t1.*, t2.*
FROM   t1 OUTER APPLY
             (
                SELECT TOP 1 *
                FROM
                    (
                        SELECT *, 'a' src FROM tb10
                        UNION ALL
                        SELECT *, 'b' src FROM tb11
                    )s
                WHERE   src = CASE WHEN @bCond = 0 THEN 'a' ELSE 'b' END
             ) t2
