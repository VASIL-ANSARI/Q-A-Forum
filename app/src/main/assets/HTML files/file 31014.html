Yes, it is definitely possible. In fact, the only problem with your code is that your lambda should accept an int. With that fix, your code compiles just fine:

int half(int x) { return x / 2; }

int main() {
    std::vector<int> test(5);
    std::transform(test.begin(), test.end(), test.begin(), [](int) { return half(4); });
    //                                                        ^^^
    return 0; 
}
