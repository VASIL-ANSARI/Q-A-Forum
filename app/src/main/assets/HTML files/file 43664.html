You need also to find the index i, which correspond to the maximum element. After that you can use cout in the for loop.

#include<iostream>;


using namespace std;


int main()

{

float array[3][2] = { { 16, 22 }, { 258, 1 }, { 42, 54 } };
int i;
int indmax;
float max = 0;

for (i = 0; i<3; i++)
{

    if(array[i][1] > max)
    {
        max = array[i][1];
        indmax=i;  
    }

}
cout << "The maximum value in the array is " << max << endl;
cout << "The row containing max values: ";
for(int j=0;j<2;j++)
    cout << array[indmax][j] << " ";   
cout << endl;  
return 0;
}
