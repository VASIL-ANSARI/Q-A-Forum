Here's a method that uses shifts to determine the numbers in the set then a loop with a modulus to repeatedly move the set into the output array. Is this the sort of simplification you were looking for?

unsigned array[17],bitmask,buf[sizeof(array[0])*8],i,pos;

bitmask = (1<<2) | (1<<7) | (1<<12);

for(i=pos=0; i<sizeof(array[0])*8; i++)
    if( (bitmask & (1<<i))!=0 ) buf[pos++] = i;

for(i=0; i<sizeof(array)/sizeof(array[0]); i++)
    array[i]=buf[i % pos];
