function populateSelect(selectId, options){

    options = options.split(';');
    selectId.find('option').remove().end();

    $.each(options, function(i, option){
      option = option.split(':');

     selectId.append($('<option>')
            .val(String(option[0])) //NOTE - no trim
            .text(String(option[1])));
    });
}
