it's because you're not setting Expander.Content to display ItemsPresenter. Your Expander should look something like this:

<Expander>
   <Expander.Header>
     <DockPanel>
       <TextBlock Text="{Binding Path=Name}" />
       <TextBlock Text="{Binding Path=ItemCount}"/>
     </DockPanel>
   </Expander.Header>
   <Expander.Content>
     <ItemsPresenter />
   </Expander.Content>
</Expander>
