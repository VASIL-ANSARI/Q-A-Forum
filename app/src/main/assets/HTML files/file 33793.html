    END 
WHEN FILE_TYPE = 'log' THEN 
    CASE
        WHEN DATABASENAME = 'tempdb' THEN 'Data, TempDB Transaction Log' 
        ELSE 'Data, Transaction Log'
    END
ELSE  ''
END
