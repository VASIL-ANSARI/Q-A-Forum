A scalar query only returns the first column of the first row
This assumes the column you want to get a max from is named id and is a long

{
   SetConnection();                        
   sql_con.Open();                    
   sql_cmd = sql_con.CreateCommand(); 

   myQuery = "Select max(id) from myTable;"; 
   Debug.WriteLine(myQuery);      
   sql_cmd.CommandText = myQuery;            
   rw_maxid = (long)sql_cmd.ExecuteScalar();
