$(document).on("click", ".acceptCustomerBut", function(e) {
    e.preventDefault();
    var $td=$(this).closest('tr').find("td:eq(1)");
    if($td.find('input').length){ // check input length, if exists then hide it
        $td.find('input').hide();
    } else { // else add it
        $td.html("<input type='number' name='customer_id'>");
    }
});
