function loadCalender(callback) {
    $.ajax({
        url: '<?php echo base_url()."catering"."/"."loadStoreCalender"; ?>',
        method: 'POST',
        data: {
            val: $(this).val()
        },
        dataType: 'json',
        context : this,
        success: function(results) {
            callback(results);
        }
    });
    return false;
}
