$(document).ready(function(){

        var a = $(this).find("td:eq(2)").find("input").val();
        var b = $(this).find("td:eq(4)").find("input").val();
        var c=parseInt(a)+parseInt(b);
        $(this).find("td:eq(5)").find("input").val(c);
    });
