function iterareIndex(index, array, iterator) {
    var previousValue = array[index];
    for (var i = 0; i < 10; ++i) {
        array[index] = i;
        if (index == array.length - 1) {
            iterator(array);
        } else {
            iterareIndex(index + 1, array, iterator);
        }
    }
    array[index] = previousValue;
}
