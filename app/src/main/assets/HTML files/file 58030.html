You can use JQuery to achieve something like that slider. I'm only using one image auto slider in the example, this can be changed for your own needs.

I didn't just want to hand you the complete answer on the spoon and feed you with it. 

$(function() {

     setInterval(function () {

       Left();

     }, 2000);



    var count = $('#slider ul li').length;

    var width = $('#slider ul li').width();

    var height = $('#slider ul li').height();

    var UlWidth = count * width;



    $('#slider').css({

        width: width,

        height: height

    });



    $('#slider ul').css({

        width: UlWidth,

        marginLeft: -width

    });



    $('#slider ul li:last-child').prependTo('#slider ul');



    function Left() {

        $('#slider ul').animate({

            left: +width

        }, 200, function () {

            $('#slider ul li:last-child').prependTo('#slider ul');

            $('#slider ul').css('left', '');

        });

    };

});
#slider {

    position: relative;

    overflow: hidden;

}

#slider ul {

    position: relative;

    margin: 0;

    padding: 0;

    height: 200px;

    list-style: none;

}

#slider ul li {

    position: relative;

    display: block;

    float: left;

    margin: 0;

    padding: 0;

    width: 500px;

    height: 300px;

    background: #ccc;

    text-align: center;

    line-height: 300px;

}

.slider_option {

    position: relative;

    margin: 10px auto;

    width: 160px;

    font-size: 18px;

}
