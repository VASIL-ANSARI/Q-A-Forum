Your function test isn't reachable because it is inside of the document.ready function closure. Put it outside of it, and you'll be able to access it normally by:

// the parent page:
$(document).ready(function () { 
    $('.loadpg').load('teste5b.php');  
});

function test () { /* your function here */ }

// inside your loaded page, you can call it like this:
window.parent.test();
