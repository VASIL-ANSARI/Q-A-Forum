I fixed it. I'm not really sure why it was concatenating the strings the way it was, but tweaking didStartElement() fixed the problem.

func parser(parser: NSXMLParser, didStartElement elementName: String, namespaceURI: String?, qualifiedName qName: String?, attributes attributeDict: [String : String]) {

        attributes = attributeDict

        element = elementName
        if ((elementName as NSString).isEqualToString("item")) {
            elements = NSMutableDictionary()
            elements = [:]
            itemTitle = NSMutableString()
            itemTitle = ""
            link = NSMutableString()
            link = ""
            thumbnailURL = NSMutableString()
            thumbnailURL = ""
        } else if ((elementName as NSString).isEqualToString("media:thumbnail")) {
            thumbnailURL.appendString(attributeDict["url"]!)
        }
    }
