Pass data from the old timepicker

$("#timeformat").change(function () {
    $past-time = $(".mytime").val();
    $(".mytime").timepicker("destroy");
    $(".mytime").timepicker({
       controlType: 'select',
       oneLine: true,
       timeFormat: $(this).val();  
    });
    $(".mytime").val($past-time);
