as requested without the empty line at the end:

def program():
    x = requestString("     ")
    print "%s\n%s\n%s" % (x,x,x)
program()

with variable number of prints:

def program():
    a_string = requestString("     ")
    a_number = 10
    print '\n'.join(a_string for x in range(a_number))
program()
