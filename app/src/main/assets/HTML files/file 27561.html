Add perfect square to a look up array and check if already it is a perfect square then ignore it.

Use:

<?php

$isAlreadySquare = array();
for($i=1; $i<=30; $i++) {

    $isSquare = false;
    for($j=1; $j<$i; $j++) {
        $s = $j*$j;
        if($s==$i && !in_array($j,$isAlreadySquare)) {
            $isAlreadySquare[] = $i;
            $isSquare = true;
        }

    }

    if($isSquare) {
        echo "square<br>";
    } else {
        echo $i."<br>";
    }
}

?>
