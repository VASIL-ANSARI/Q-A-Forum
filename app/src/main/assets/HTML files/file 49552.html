def name_get(self, cr, uid, ids, context={}):
    if not len(ids):
        return []

    res=[]

    for assest in self.browse(cr, uid, ids,context=context):
        res.append((assest.id, assest.asset_catg_id + ', ' + assest.folio_num))     

    return res
