You can use flexbox for that:

.parent {

  width: 50%;

  height: 200px;

  border: 1px solid grey;

  margin: 1em auto;

  display: flex;

  flex-direction: column;

  align-items: center;

}

.child {

  width: 60%;

  height: 50px;

  background: rebeccapurple;

  margin-top: auto;

}
<div class="parent">

  <div class="child"></div>

</div>
