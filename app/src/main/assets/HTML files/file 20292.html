Use the stop(true) before each animation, like this:

$(".slidingDiv").stop(true).slideToggle();

Use this instead and let me know if it works:

$(document).ready(function()
{
  $(".slidingDiv").hide();
  $(".show_hide").show();

  $('.show_hide').hover(
      function () {
         $(".slidingDiv").stop(true).slideToggle();
      },
      function() {
         $(".slidingDiv").stop(true).slideToggle();
      }
  );

});
