In your ajax request you should specify the dataType as "html". Among others : when data is generated by php json_encode() function.

 $.ajax({
            type : "GET",
            url : "data/data.json",
            dataType : "html",
            data : data,                    
            success:function(msg){
                console.log(msg);
                var array_return =  $.parseJSON ( msg );
            },
            error:function(msg){
                console.log("failed");
            }
        });
