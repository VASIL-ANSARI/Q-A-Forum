IE is notorious for caching ajax requests. Try adding the parameter 'cache: false' to your request:

$.ajax({
    type: 'POST',
    cache: false,
    url: 'ajax.php',
    data: 'some=info',
    dataType: 'json',
    success: function(data) {
        if(data != null) {
            console.log(data);
        } else {
            alert('IE fails here!');
        }
    }
});
