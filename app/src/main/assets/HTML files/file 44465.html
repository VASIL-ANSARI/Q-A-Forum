Because you are redefining rows array in every iteration.

Try this:

var ids = [];
var names = [];

for (var i = 0; i < rows.length; i++) {
    ids.push(rows[i].EquipmentID);   
    names.push(rows[i].Name);  
}

socket.emit('Switch', {eqid: ids, eqroom: names});
