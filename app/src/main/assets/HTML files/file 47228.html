import pandas as pd

df = pd.DataFrame([
                   ['Delhi',  'Doctors',  500],
                   ['Delhi',  'Lawyers',  400],
                   ['Delhi',  'Plumbers', 100],
                   ['London', 'Doctors',  800],
                   ['London', 'Lawyers',  700],
                   ['London', 'Plumbers', 300],
                   ['Tokyo',  'Doctors',  900],
                   ['Tokyo',  'Lawyers',  800],
                   ['Tokyo',  'Plumbers', 400]],
                  columns = ['City',   'Job',      'Salary'])

print df.groupby('Job').Salary.median().to_frame('salary_median')
