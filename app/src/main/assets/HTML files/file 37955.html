SELECT
     `t1`.*
    ,`t2`.*
FROM `t1`
LEFT JOIN `t2` ON `t1`.websiteID = `t2`.ownerID
    AND `t1`.authorID = `t2`.authorID
    AND `t1`.authorID = 1559
WHERE ( `t1`.authorID = 1559 OR `t1`.websiteID = 1559 )
    AND `t1`.postType = 'simplePost'
ORDER BY postID limit 0 ,50
