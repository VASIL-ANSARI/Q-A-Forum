FROM `cat_materii` --`cat_rapoarte` 
INNER JOIN `cat_rapoarte` on `cat_materii`.`m_id`=`cat_rapoarte`.`rap_m_id` 
WHERE `cat_rapoarte`.`k_id` = '7fbXe1dvltedEkIXELc8Q1NeMkKRb3pi'
AND (data BETWEEN '2015-11-01' AND '2015-11-30') 
GROUP BY `rap_m_id`
