CREATE PROCEDURE `follow`(IN in_follower INT(11), IN in_followee VARCHAR(45))
    INSERT INTO follower 
    (follower_id, followee_id)
    SELECT in_follower, `user_id` 
    FROM   user 
    WHERE  `username` = @in_followee);
END
