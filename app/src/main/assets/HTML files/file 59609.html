You use while True so after you close window while True opens new window.

Use last line menu1() without while True

EDIT:

from tkinter import *

def quit2():
    menu.destroy()

def menu1():
    global menu

    menu = Tk()

    play = Button(menu, text='play', command=main)
    play.pack()
    quit1 = Button(menu, text='quit', command=quit2)
    quit1.pack()

    menu.mainloop()

#without `while True`
menu1()
