$result = preg_match_all('/(--[0-9]+)|(GET|PUT|POST)(.*)HTTP|Accept:(.*)|content-type:(.*)|(\"\{.*)/mi', $data, $matches);

$verbs = $matches[2];
$urls = $matches[3];
$accepts = $matches[4];
$contentTypes = $matches[5];
$bodies = $matches[6];
$separators = $matches[1];

$final = [];

function extractData(&$final, $verbs, $urls, $accepts, $contentTypes, $bodies, $separators, $start)
{
    $next = $start;
    if ($start >= count($verbs)) {
        return;
    }

    if ('' !== $verbs[$start]) {
        $indexFinal = count($final);
        $final[$indexFinal]['method'] = $verbs[$start];
        while ($separators[$next] === '' && $next < count($separators)) {
        if ($urls[$start] !== '') {
            $final[$indexFinal]['url'] = $urls[$start];
        }

        if ($accepts[$next] !== '') {
            $final[$indexFinal]['accept'] = $accepts[$next];
        }

        if ($contentTypes[$next] !== '') {
            $final[$indexFinal]['content-type'] = $contentTypes[$next];
        }

        if ($bodies[$next] !== '') {
            $final[$indexFinal]['body'] = $bodies[$next];
        }
        $next++;
    }
        extractData($final, $verbs, $urls, $accepts, $contentTypes, $bodies, $separators, $next);
    } else {
        extractData($final, $verbs, $urls, $accepts, $contentTypes, $bodies, $separators, $start + 1);
    }
}

extractData($final, $verbs, $urls, $accepts, $contentTypes, $bodies, $separators, 0);
