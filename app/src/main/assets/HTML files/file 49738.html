As long as it isn't a HTTP only cookie (http://blog.codinghorror.com/protecting-your-cookies-httponly/), it should be easily readable with javascript, see: "Shortest function for reading a cookie in JavaScript" for code on how to read a cookie in javascript, or use a plugin. If JSESSIONID is server only, then you will have to write or find a plugin that can intercept the network call to retrieve the data from the headers.

Edit: Getting the cookie

function readCookie(name) {
    var allCookies = document.cookie.split('; ');
    var cookies = {};

    for (var i = allCookies.length - 1; i >= 0; i--) {
        var splitCookie = allCookies[i].split('=');
        cookies[splitCookie[0]] = splitCookie[1];
    }

    return cookies[name];
}
