For UI Text, you use Text.color not Renderer.material.color. Renderer is used for 3D Objects with Mesh Renderer attached to them. Make sure to attach this to a GameObject with a Text Component attached to it or you will get an error.

public float timer = 0.0f;
Text txt;

void Start()
{
    txt = gameObject.GetComponent<Text>();
}


void Update()
{
    timer += Time.deltaTime;
    if (timer >= 2.0f)//change the float value here to change how long it takes to switch.
    {
        // pick a random color
        Color newColor = new Color(Random.value, Random.value, Random.value, 1.0f);
        // apply it on current object's material
        txt.color = newColor;

        timer = 0;
    }
}
