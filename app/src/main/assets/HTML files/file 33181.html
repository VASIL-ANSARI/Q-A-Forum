function TweetData() {

    this.tweetList = [];
    var tweet=this;

    var data = $.getJSON('data/tweets.json'); 

    data.done(function(data){

        for (var i = 0; i < data.length; i++) {
            var obj = data[i];
            // 'tweet' is created outside of the callback
            // and is then passed into the data.done promise
            tweet.tweetList.push(obj.text);
        }

    });

} //end constructor
