    session_start()

// Only save the page if the user has not joined before.
if (!isset($_SESSION['has_joined_before'])) {

    // Get the URL
    $first_page = "http://".$_SERVER[HTTP_HOST].$_SERVER[REQUEST_URI];

    // Don't save the first page again.
    $_SESSION['has_joined_before'] = true;
