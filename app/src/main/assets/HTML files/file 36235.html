def upload(request):
    if request.method == 'POST':
        cursor = connection.cursor()
        x=[]
        reader = csv.reader(request.FILES['csvData'],delimiter=',')
        i = 0
        for row in reader:
            obj = new MyObject()
            obj.country = row[0]
            obj.city = row[3]
            ...
            x.append(obj)
            if i>=5000:
                MyObject.objects.bulk_create(x)
                x=[]
                i=0

            i = i+1
