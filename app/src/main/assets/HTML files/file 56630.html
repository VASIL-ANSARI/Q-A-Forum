To avoid excess copies/moves, use forward_as_tuple for the temporary. That returns a tuple of references, which can be readily optimized away by any decent optimizer.

template <std::size_t... Is, class... Ts>
auto custom_expand( std::index_sequence<Is...>, Ts&&... args )
{
    auto tmp_tuple = std::forward_as_tuple( std::forward<Ts>( args )... );
    return std::make_tuple( std::get<Is>( std::move(tmp_tuple) )... );
}
