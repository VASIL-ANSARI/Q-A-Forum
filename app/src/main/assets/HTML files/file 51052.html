I have simplified your javascript a bit and kept your HTML for reference, have a look at this: https://jsfiddle.net/zzu22njs/

Javascript:

var $popup = $('.popScroll')

$popup
  .find('a.close')
  .on('click', function(e) {
    e.preventDefault();
    $popup.hide();
  });

var now = new Date();

// Open on Fridays
if (now.getDay() === 5) {    
  $popup.show();
}

CSS:

.popScroll {
  display: none;
}
