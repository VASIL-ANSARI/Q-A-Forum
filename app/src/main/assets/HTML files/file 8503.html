$xpath = new DOMXPath($xml);
$ingdiv = $xpath->query("/recipeml/recipe/ingredients/ing-div");
$length = $ingdiv->length;
// iterate over all <ing-div> from last to first
while($length--)

    print $xpath->query("ing", $ingdiv->item($length))->length;
}
