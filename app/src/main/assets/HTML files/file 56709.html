if you are using java6 or java7 you can get javascript variable in java like below :

(NativeObject has been removed in java8 but there is a similar class [jdk.nashorn.api.scripting.ScriptObjectMirror] you can use)

public class Test5 {
    private static final String TEST_SCRIPT =
            "var employees = {"
            + " accounting: []"
            + "};"
            + "for(i=1; i<=3; i++) {"
            + " employees.accounting.push({ "
            + " \"firstName\" : \"firstName\" + i,"
            + " \"lastName\"  : \"lastName\" + i,"
            + " \"age\"       : \"age\" + i"
            + "});"
            + "}"
            + "employees;";
    /**
     * @param args
     */
    public static void main(String[] args) {
        try{
            CompiledScript script = compEngine.compile(TEST_SCRIPT);
            NativeObject obj = (NativeObject) script.eval();
            NativeArray obj1 = (NativeArray) obj.get(obj.getAllIds()[0]);
            for (Object temp : obj1) {
                NativeObject obj2 = (NativeObject) temp;
                System.out.println(obj2.get("firstName"));
                System.out.println(obj2.get("lastName"));
                System.out.println(obj2.get("age"));
            }
        }catch(Exception e){e.printStackTrace();}
    }
}
