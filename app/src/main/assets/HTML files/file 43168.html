It's because you are concatenating one long string. Try to use array to hold values:

var i;
var vxpResult = [];

for (i = 0; i < VXP.length; i++) {
    startingValue *= VXP[i];
    vxpResult.push(Math.floor(startingValue));
}

function MyCtrl($scope) {
    $scope.level = i;
    $scope.vxp = vxpResult;
}
