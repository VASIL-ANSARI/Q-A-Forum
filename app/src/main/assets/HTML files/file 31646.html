If you don't have .Net Framework 4.0, then substitute Tuple with non-generic version (declare list as List<Tuple>):

class Tuple
{
    public string Item1;
    public string Item2;
    public string Item3;
    public Tuple(string i1, string i2, string i3)
    {
        Item1 = i1;
        Item2 = i2;
        Item3 = i3;
    }
}
