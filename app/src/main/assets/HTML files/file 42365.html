I came up with this solution.Please note that Kendo classes are not imported in this example so it wont show the correct result. Please import kendo css classes to get the correct results as shown below in the screenshot. 

$("#tree").kendoTreeView({

    select: function(e) {

        var breadcrumb = [];

        var treeView = this;



        $(e.node)

            .parents("li")

            .andSelf()

            .each(function() {

                var nodeText = $.trim(treeView.text(this));

                breadcrumb.push(nodeText);

            });



        breadcrumb = breadcrumb.join(" > ");



        $("#breadcrumb").text(breadcrumb);

        

    }

    

});
