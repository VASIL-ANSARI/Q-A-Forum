you need to pass data to the open function of the dialog box like below code, and you can use data- attributes to pass the values. jquery Data() method will read the values. 

$('#row').click(function()
  {
     $("#dialog").data('id',$(this).data('row')).dialog('open'); 
  });

 $('#not_row').click(function()
  {
     $("#dialog").data('id',$(this).data('row')).dialog('open'); 
  });
