<script>
$(document).ready(function() {
    $('#search').keypress(function(){
        $.ajax({

            type: "GET",
            url: "plan-api.php",
            data: 'result=' + $('#search').val(),
            dataType: "json",

            success: function(responseText){
                $.each(responseText.data, function(key,value){
                    $('#result'+key).val(value.recharge_amount);
                });
            }

        }); // Ajax Call
    }); //event handler

}); //document.ready
</script>
