You will have but two options here: clone the CodeIgniter library or use ajax. The ajax option would be easy by using jQuery.get() and some controller actions specifically coded for ajax, usually just a JSON string response back to jQuery, but that is some work! The cloning idea is a great one actually. Your library would be like this:

class Test_Controller extends CI_Controller {

public function page(){

    $this->load->library('pagination');

    $paginationA = clone($this->pagination);
    $paginationA->initialize(array('uri_segment' => 3));

    $paginationB = clone($this->pagination);
    $paginationB->initialize(array('uri_segment' => 4));

}

}
