Assuming the login name is "Rose", this should work (untested)

-- Remove login from sysadmin
EXEC master.dbo.sp_dropsrvrolemember
     @loginame = 'Rose'
   , @rolename = 'sysadmin'
;
GO

-- [ab]use the undocumented procedure to loop over all databases
EXEC sp_msforeachdb '
  USE [?];

  -- Create the user
  CREATE USER "Rose" FOR LOGIN "Rose";

  -- Assign the db_owner role
  EXEC sp_addrolemember
       @rolename   = ''db_owner''
     , @membername = ''Rose''
  ;
';

Give it a quick test before you put it in to production!
