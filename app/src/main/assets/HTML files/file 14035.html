Assuming your RDMS doesn't support window functions,

SELECT a.*
FROM table1 a
INNER JOIN 
(
  SELECT name,MAX(date) as max_date -- date field may need escaping
  FROM table1 a
  GROUP BY name
)b ON (b.name = a.name AND a.date=b.max_date)
