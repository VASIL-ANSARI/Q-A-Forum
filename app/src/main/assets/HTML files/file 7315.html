$('.ddm li').click(function() {
    var me = $(this);
    var txt = me.text();
    var id = me.parent().attr('id');
    $('.ddm').not('#' + id + '').each(function(i) {
    $('li:contains(' + txt + ')', this).remove();
    });
});
