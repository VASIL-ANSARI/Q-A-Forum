Can you just skip over zeroes? Also, you might as well put all the calculations inside a single loop.

int n = 0 // number of data points

for (int i = 0; i < 139; ++i) {
  if (lag[i] == 0) {
    continue;  // skip 0 values
  }
  ++n;
  sum += lag[i];

  if (highest < lag[i]) {
    highest = lag[i];
    highestWorld = i + 1;
  }

  if (lowest > lag[i]) {
    lowest = lag[i];
    lowestWorld = i + 1;
  }

  average = sum / n;  // Note: you may want to round this.
}
