If You want to distinguish between them, use success(data, textStatus, jqXHR) event and read it from jqXHR.status

or access jqXHR other way:

var jqxhr = $.ajax(
    ...     
    statusCode: {    
        200: function() {    
            if(304 == jqxhr.status)
                alert("not modified"); // does not fire
        }    
    },    
)
