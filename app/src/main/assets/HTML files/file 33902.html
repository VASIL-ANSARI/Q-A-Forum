$(function() {
    var startTime = localStorage.getItem("startTime");
    if (startTime == null) {
        startTime = new Date().getTime();
        localStorage.setItem("startTime", startTime);
    }
    var countdown = setInterval(function() {
        var elapsed = Math.floor((new Date().getTime() - startTime) / 1000);
        var remaining = 60 - elapsed;
        $("span.countdown").html(remaining + " seconds Remaining");
        if (remaining == 0) {
            clearInterval(countdown);
            localStorage.removeItem("startTime");
        }
    }, 1000);
});

Store the start time and then calculate the time elapsed since then. If you want to get really messy then you can also calculate the time between pages so that you get 60 seconds of "on page" time, rather than 60 seconds that includes page transitions.

Either way, that will solve your immediate issue of the timer not going down.
