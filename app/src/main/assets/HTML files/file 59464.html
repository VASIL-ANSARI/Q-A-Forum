The example on that question is pretty good. You don't need the focus. You just need to call the button "action" function.

Here is the solution anyway, for that same example linked above:

var presscount = 0;
var sendEvery = 5000; //in milliseconds
$(document).ready(function() {
  $("#test").on("keypress", function(e) {
    if (e.which == 32) {
      presscount++;
    }
    $("#output").text("The space key was pressed " + presscount + " times");
  });
     setInterval(pressKey, sendEvery);  
});

function pressKey() {
  var e = jQuery.Event("keypress");
  e.which = 32; // # space
  $("#test").trigger(e);
}
