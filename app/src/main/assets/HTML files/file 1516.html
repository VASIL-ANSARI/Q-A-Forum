How about just replacing anything within braces with a span, then setting the html to the replaced text and simply copying the existing link into the span? I think you'll need to quote the braces as they are special in a regular expression -- I could be wrong on this, though, for javascript. If so, let me know and I'll update. Note the use of closest and find with first to get the first link in the same row as the current table element.

$(function() {
    $('td.german').each(function() {
        var link = $(this).closest('tr').find('a:first').clone();
        var re = new RegExp( '\{.*\}' );
        var sentence = $(this).text();
        var word =  sentence.match(re)[0];
        word = word.substr(1,word.length-2);
        link.text(word);
        sentence = sentence.replace(re,'<span></span>');
        $(this).html(sentence).find('span').append(link);
    });
});
