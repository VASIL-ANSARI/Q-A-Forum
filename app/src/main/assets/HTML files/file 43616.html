It should be data: {'client' : $("#client").val()} instead of "data: 'client' : $("#client").val()". Look at this example. So code which is below should work properly:

$(document).ready(function(){
      $.ajax({
           url: 'jsonClients.php',
           type: 'POST',
           dataType: 'json',
           data: {'client' : $("#client").val()},
           success: function(data){
                 $('#client').autocomplete(
                 {
                       source: data,
                       minLength: 1
                 });
           }
      });
});
