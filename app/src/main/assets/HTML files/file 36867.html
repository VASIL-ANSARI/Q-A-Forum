<?php

class C {
    protected function C1()
    {
        return 'c1';
    }
}

class B extends C 
{
    protected function B1()
    {
        return 'b1';
    }
}

class A extends B 
{
    public function __construct()
    {
        $var1=$this->B1();
        $var2=$this->C1();
        echo 'var1: '.$var1.'<br>var2: '.$var2;
    }
}

new A;

?>
