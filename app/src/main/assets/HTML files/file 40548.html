You can use a window function

WITH T AS
(
SELECT *,
       ROW_NUMBER() OVER (PARTITION BY id ORDER BY Logged DESC) AS RN
FROM Resources
)
SELECT *
FROM T
WHERE RN =1

Or another way might be

WITH T AS
(SELECT MAX(Logged) AS Logged,
        ID
FROM Resources
GROUP BY ID
)
SELECT R.*
FROM Resources R
JOIN T ON T.ID=R.ID AND T.Logged = R.Logged
