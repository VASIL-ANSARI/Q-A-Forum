function showDiv(element){
    var $img = $(element).stop().animate({"height": "toggle"}, { duration: 1000 }).prevAll('img');
    if ($img.attr('src') == 'icon/down.png') {
        $img.attr('src', 'icon/up.png');
    } else {
       $img.attr('src', 'icon/down.png');
    }
}
