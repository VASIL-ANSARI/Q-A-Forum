You can pass an array of parameters to bind to the select.

 class Webmaster {
      public static function webmasters($filt, $cat) {
         $top_pages = DB::select(DB::raw("SELECT *
                                         FROM web.tools
                                         WHERE filter = :filter
                                         AND category = :category
                                         LIMIT 20"), [
                                             ':filter' => $filt,
                                             ':category' => $cat
                                         ]);
          return $top_pages;
       }
 }
