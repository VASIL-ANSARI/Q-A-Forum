You can access all child-views (like UIButtons) which are in this view (if they are in other view in this view you will need to use recurency) by accessing UIView property subviews.

//UIView
@property(nonatomic, readonly, copy) NSArray *subviews

like this:

NSArray *subviews = myView.subviews;

for(NSObject *uiobject in subviews){
    if([uiobject isMemberOfClass:[UIButton class]]){
       // alter your buttons here
    }//if there are for example uiViews you can try use recurrency 
}
