List supports those features and is O(1) for add if the capacity > count

List.Add Method

Everything else is O(N) but in practice is better.
If you inset at position 0 then it is O(N) but if you inset at the second to last you will find it is better than O(N).

    List<simple> ls = new List<simple>();
    simple ss = new simple(0);
    ls.Add(ss);
    ls.Add(new simple(2));
    ls.Insert(0, new simple(1));
    ls.Add(new simple(3));
    ls.Add(new simple(4));
    ls.RemoveAt(2);
    foreach(simple s in ls) 
    {
        System.Diagnostics.Debug.WriteLine(s.ID);
    }
    System.Diagnostics.Debug.WriteLine(ls[1].ID.ToString());
    System.Diagnostics.Debug.WriteLine(ls.IndexOf(ss).ToString());
}
public class simple
{
    public Int32 ID { get; private set; }
    public simple(Int32 id) { ID = id; }
}
