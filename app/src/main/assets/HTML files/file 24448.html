If you are using Node.js you can basically do something like this, make sure you have the express and less NPM packages installed:

var express = require('express');
var less = require('less');
var app = express();

app.get('/getCSS/:id', function(req, res){
  less.render('.class { width: ' + req.params.id  + 'px; }', function (e, css) {
    res.send(css);
  });
});

app.listen(3000);

This is a quick 'n dirty script (no error checking or input validation whatsoever) that will accept requests to http://localhost:3000/getCSS/:id and output the given id as some variable in your LESS code.

# curl http://localhost:3000/getCSS/4
.class {
  width: 4px;
}

Of course you are free to use the parameters any way you like by inserting them into your LESS templates.
