Personally I handle this with one method and a global variable.

Example:

var backState = false;

function backButton() {
    if(backState === 'LogicA') {
       //do logic a
    }
    elseif(backState === 'LogicB') {
       //do logic b
    }
    else {
       //some default action if needed
    }
}

document.addEventListener("backbutton", backButton, false);

Then as needed I can change the value of backState and directly affect the function of the back button.
