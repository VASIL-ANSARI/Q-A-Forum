This is what your query would look like with a subquery:

SELECT
CONCAT(m.Nume,' ',m.Prenume) AS Medic,
  m.Specialty,
COUNT(v.Data_viz) AS nrviz
FROM
  visits v
INNER JOIN
  medics m
ON(v.Medic_id=m.id_med)
WHERE
YEAR(v.Data_viz) = (SELECT...)
GROUP BY
  Medic
ORDER BY
  m.Specialitate,
  nrviz DESC
