Use $elemMatch projection to retrieve the array. The mongo query which you want to generate isFinite

db.project.findOne(
    {"users.id": "58b5eb81-35b2-4719-87ad-e5eea76a478f"},
    {
        "_id": 0,
        "users": {
            "$elemMatch": { "id": "58b5eb81-35b2-4719-87ad-e5eea76a478f" }
        }
    }).users[0]

or use the $ projection operator which projects the array elements based on some condition from the query statement. In your case your query statement will use the dot notation to query for the Images array where it has at least an element:

db.project.findOne(
    {"users.id": "58b5eb81-35b2-4719-87ad-e5eea76a478f"},
    { "_id": 0, "users.$": 1 }
).users[0];
