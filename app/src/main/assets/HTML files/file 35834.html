There are other issues with the code I have written, but I have solved this particular issue by changing this section:

var inputval = $(".name").map(function() {
    p = $(this).val();
    return p;
}).get();

to this:

var inputval = $(".name").map(function() {
    if (!$(this).is("select")) {    
        p = $(this).val();
    } else {
        p = $(this).find("option:selected").text();
    }
    return p;
}).get();
