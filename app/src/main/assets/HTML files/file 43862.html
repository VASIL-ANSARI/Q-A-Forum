I would take a modular approach. Create a function that calls editable with default properties, which are extended with properties specific to each element.

Something like this:

function setup(obj, options) {
  var defaults= {
        type: 'text',
        placement: 'right',
        params: {
          controller:'detail',
          cid:ayb.component.id,
          CSRF:ayb.CSRF
        },
        url: 'index.php',
        pk: function(){return $('#id').val();},
        validate: function(value) {
                    if($.trim(value) == '') {
                      return 'This field is required';
                    }
                  }
        }
  obj.editable($.extend(true, defaults, options));
} //setup

setup($('#phone'), {
        title: 'Phone Number',
        inputclass: 'phone',
        params: {
          task: 'savePhone'
        }
      }
     );

setup($('#fax'), {
        title: 'Fax Number',
        inputclass: 'phone',
        params: {
          task: 'saveFax'
        }
      }
     );
