SELECT 'Title', 'Date', 'Count'
UNION ALL
SELECT Title,Date,Count
FROM sales_com
WHERE Date >= to_date('2004/01/01', 'yyyy/mm/dd')
AND Date <= to_date('2004/12/31','yyyy/mm/dd');
INTO OUTFILE 'U:/CSV-LiveDB-Upload/sales_com/Data.csv'
FIELDS TERMINATED BY ','
ENCLOSED BY '\"'
LINES TERMINATED BY '\r\n';
