You need to parse the JSON object before trying to pull out the elements. You do this by using JSON.parse()

<script>

$(document).ready(function(){
  $("#asd").click(function(){
    $.post("/sister/rest/rest/login",
    {
      email:"t@s.com",
      pwd:11
    },
    function(data){
      data = JSON.parse(data);
      $('#id3').append(data.user_fullname);
    });
  });
});

</script>
