#include <iostream>
#include <fstream>
#include <string>
#include <cstring>
using namespace std;

int main() {
  char c;
  char arr[100];
  size_t i = 0;
  // read data from file
  ifstream myfile;
  myfile.open("filename.txt", ios::in);
  while (!myfile.eof()) {
    myfile >> c;
    if (isdigit(c) || c == '.') {
      arr[i] = c;
      i++;
    }
  }
  arr[i] = '\0';  // mark end of array
  myfile.close();  // file closed
  size_t k = strlen(arr);
  // parsing string
  int newarr[100];
  int m = 0;
  string s;
  for (i = 0; i <= k; i++) {
    s.erase();
    while (arr[i] != '.') {
      s += arr[i++];
    }
    newarr[m++] = atoi(s.c_str());
  }

  newarr[m] = '\0';
  // printing newarray
  i = 0;
  while (newarr[i] != '\0') {
    cout << newarr[i++] << '\t';
  }
  return 0;
}
