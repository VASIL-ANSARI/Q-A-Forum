This should give you a start. Anchor each of the divs to its corner, then animate height, width and opacity:

$('button').click(function() {

  $('.animate').animate({'opacity' : 0,

    'height': 0,

    'width': 0

  }, 1000);

})
.animate {

  width: 50%;

  height: 50%;

  position: absolute;

}

#topL {

  top: 0;

  left: 0;

}

#topR {

  top: 0;

  right: 0;

}

#bottomL {

  bottom: 0;

  left: 0;

}

#bottomR {

  bottom: 0;

  right: 0;

}
<script src="https://ajax.googleapis.com/ajax/libs/jquery/2.1.1/jquery.min.js"></script>

  <div class="animate" id="topR">some text here</div>

  <div class="animate" id="bottomL">some text here</div>

  <div class="animate" id="bottomR">some text here</div>

</div>
