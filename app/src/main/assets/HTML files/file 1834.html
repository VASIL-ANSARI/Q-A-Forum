Not sure that is possible, but why not use an intermediate router?

Router::connect (
    '/my/*', 
    array (
        'controller' => 'my_router', 
        'action' => 'route',
    )
);

class MyRouterController extends AppController {
    ... 
    function route ()
    {
        $args = func_get_args ();
        $controller = array_shift ($args);
        $this->requestAction (
            'my_'.$controller.'/'.implode('/', $args), 
            array ('return' => true)
        );
    }
}
