@IBAction func longPress(gesture:UILongPressGestureRecognizer) {
    let view = gesture.view!
    print("state = \(gesture.state.rawValue)")
    if gesture.state != UIGestureRecognizerState.Ended && gesture.state != UIGestureRecognizerState.Cancelled {
        view.tag = 1
    } else {
        view.tag = 0
    }
    if button1.tag == 1 && button2.tag == 1 {
        print("pressed both buttons for 3 seconds")
    } else {
        print("not pressed both buttons for 3 seconds")
    }
}

The idea is to keep a reference to both the buttons and then use a long gesture recogniser on those buttons connect to a single action. Inside the action you get the button on which the gesture is done and then change its tag to 1. If the gesture ends or is cancelled then it is set to 0. When both the buttons are pressed then both buttons will have a tag of 1 in which case you will know that the buttons have been pressed for more than 3 secs. Note that you have to set the min duration in storyboard and keep outlets to the buttons.

You can use a different condition instead of tags to check the long pressed state if required but the same idea can be used.

In ObjC

-(IBAction)longPress:(UILongPressGestureRecognizer*) gesture {
    UIButton* button = (UIButton*)gesture.view;
    if (gesture.state != UIGestureRecognizerStateEnded && gesture.state != UIGestureRecognizerStateCancelled)
    {
        button.tag = 1;
    } else {
        button.tag = 0;
    }
    if (button1.tag == 1 && button2.tag == 1) {
        NSLog(@"pressed both buttons for 3 seconds");
    } else {
        NSLog(@"not pressed both buttons for 3 seconds");
    }
}
