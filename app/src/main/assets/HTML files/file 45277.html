You can store the selected date in another field, and use the onSelect event to check and overwrite the selected value. 

$("#datepicker").datepicker({

    altField: "#actualDate",

    onSelect: function(dateText, inst) {

        if (dateText.substring(0, 5) == "04/01") {

            inst.input.val("april fools!");

        }

    }

});
