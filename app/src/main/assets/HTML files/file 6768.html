App::import('Core', 'HttpSocket');
$HttpSocket = new HttpSocket();
$input = $HttpSocket->get('http://www.example.com/something.xml');

App::import('Xml'); 
$xml = new Xml($input);
$xmlAsArray = $xml->toArray();

foreach($xmlAsArray as $item) {
    $this->Article->create();
    $data['Article'] = array(
        'title' => $item['title'],
        'contents' => $item['contents'],
        'date' => $item['date']
    );
    $this->Article->save($data);
}
