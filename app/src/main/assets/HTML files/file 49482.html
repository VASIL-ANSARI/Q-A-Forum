from your footer just call a new controller action:

{{ render(controller('AppBundle:Default:social')) }}

The new controller action:

# src/AppBundle/Controller/DefaultController.php

namespace AppBundle\Controller;

use Symfony\Bundle\FrameworkBundle\Controller\Controller;

class DefaultController extends Controller
{
    public function socialAction()
    {
        $em = $this->getDoctrine()->getManager();

        $socials = $em->getRepository('AppBundle:Social')->getAll();

        return $this->render('default/social.html.twig', array(
            'socials' => $socials,
        ));
    }

and default/social.html.twig:

#{# app/Resources/views/default/social.html.twig #}

<ul>
    {% for social in socials %}
        <li>{{ social.name }}</li>
    {% endfor %}
</ul>
