You can keep a copy of the original index and reindex to that, thanks Andy Hayden.

Demo:

#              A  B
#first second      
#zzz   z       2  4
#      a       1  5
#aaa   z       6  3
#      a       7  8

print df.index
#MultiIndex(levels=[[u'aaa', u'zzz'], [u'a', u'z']],
#           labels=[[1, 1, 0, 0], [1, 0, 1, 0]],
#           names=[u'first', u'second'])

#set index to variable
index = df.index

#stack and unstack
df = df.unstack().stack()
print df
#              A  B
#first second      
#aaa   a       7  8
#      z       6  3
#zzz   a       1  5
#      z       2  4
#              A  B

df = df.reindex(index)
print df
#              A  B
#first second      
#zzz   z       2  4
#      a       1  5
#aaa   z       6  3
#      a       7  8
