$("#div1").hide();
$("#div2").hide();
$('#radiobutton1').click(function() {     
    var checked = $(this).attr('checked', true);
    if(checked)
    { 
        $("#div1").show();
    }
    else
    { 
        $("#div1").hide();
    }
});
