SELECT ru.id, ru.description, re.result_of_the_rule, re.date
    FROM (SELECT rule_id, MAX(date) AS MaxDate
              FROM result
              GROUP BY rule_id) q
        INNER JOIN rules ru
            ON q.rule_id = ru.id
        INNER JOIN result re
            ON q.rule_id = re.rule_id
                AND q.MaxDate = re.date
