Use timestamps in your mysql table and then make another php file for checking the latest timestamp with your conditions. check for this timestamp and save the timestamp of the latest db-update in a javascript variable.

var latest_timestamp = 0;

setInterval(function () {

    $.ajax({
        url: "check_timestamp.php",
        type: "GET",
        success: function(latest_db_timestamp){
            if(latest_timestamp < latest_db_timestamp)
            {
                $('#respnseDiv').load('response.php');
            }
        }
    });

}, 3000);
