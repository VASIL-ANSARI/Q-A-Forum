hash[letter] = 1 will always execute. You can restructure your code like this:

letters.each do |letter|
    hash[letter] = 0 unless hash[letter]
    hash[letter] += 1
end

or without suffix conditionals:

letters.each do |letter|
    if hash[letter]
      hash[letter] += 1
    else
      hash[letter] = 1 
    end
end
