Use !$(event.target).is("textarea") jsBin demo
Also use event.which (instead of event.keyCode; jQuery normalizes it for xBrowser)

$(function() { // DOM ready

    $(window).keydown(function(event){
        if(event.which==13 && !$(event.target).is("textarea")) {
            event.preventDefault();
            console.log("ENTER PREVENTED");
            return false;
        }else{ // just to test
            console.log("ENTER was in textarea");
        }
    });

});
