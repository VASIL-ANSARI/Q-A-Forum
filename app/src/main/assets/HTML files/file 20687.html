Simplest answer:

$stmt = $pdo->prepare('SELECT * FROM ' . $table . ' WHERE id = :p');

$stmt->execute(array(':p' => $_GET['p']));

while ($row = $stmt->fetch(PDO::FETCH_ASSOC))
{
    echo $row['id'];
}
