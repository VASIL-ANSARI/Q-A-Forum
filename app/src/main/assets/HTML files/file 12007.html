$('img#inner_images').each(function()
   {
      var w1=$(this).width();
      var h1=$(this).height();


     if(h1>w1)
     {
        w1=Math.ceil(w1/h1*max_h);
        h1=max_h;
      }
       if(w1>h1)
      {
         h1=Math.ceil(h1/w1*max_w);
          w1=max_w;
      }
    $(this).css({height:h1, width:w1});
   });
