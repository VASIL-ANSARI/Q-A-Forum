Specify dataType:"html" in ajax request as shown.

function SetMark() {
    $.ajax({
         url: '<%=Url.Action("SelectMark", "Home")%>',
         type: 'POST',
         dataType:'html',
         data: {anio:$("#anio").val()},
         success: function (data) {
             $("#marca").html(data);
         }
    });
}
