internal static void enableToolbar()
    {
        ShellWindows shellWindows = new ShellWindows();
        foreach (InternetExplorer ie in shellWindows)
        {
            String filename = ie.Name;
            if (filename.Contains("Internet Explorer"))
            {
                ie.ToolBar = 1;
            }
        }
    }
