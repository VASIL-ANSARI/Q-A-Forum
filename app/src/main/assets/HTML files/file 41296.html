It seems that you can't use multiselect: true together cell editing. It you do need that you will have to use beforeSelectRow callback to fix the problem.

The demo demonstrates the solution. It uses the following implementation of beforeSelectRow:

beforeSelectRow: function (rowid, e) {
    var $self = $(this), iCol, cm,
        $td = $(e.target).closest("tr.jqgrow>td"),
        $tr = $td.closest("tr.jqgrow"),
        p = $self.jqGrid("getGridParam");

    if ($(e.target).is("input[type=checkbox]") && $td.length > 0) {
       iCol = $.jgrid.getCellIndex($td[0]);
       cm = p.colModel[iCol];
       if (cm != null && cm.name === "cb") {
           // multiselect checkbox is clicked
           $self.jqGrid("setSelection", $tr.attr("id"), true ,e);
       }
    }
    return false;
}
