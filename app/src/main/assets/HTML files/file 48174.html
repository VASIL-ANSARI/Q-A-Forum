Change your server code to send the session cookie automatically with an expiration time:

// express/mongo session storage
app.use(express.session({
  secret: 'MEAN',
  store: new mongoStore({
    url: config.db,
    collection: 'sessions'
  }),
  cookie: { maxAge : 3600000 } //1 Hour <-----------
}));

or change the passport.authenticate function to send the cookie directly with expiration time.

https://github.com/DaftMonk/angular-passport/blob/master/lib/controllers/session.js

exports.login = function (req, res, next) {
  passport.authenticate('local', function(err, user, info) {
    var error = err || info;
    if (error) { return res.json(400, error); }
    req.logIn(user, function(err) {
      if (err) { return res.send(err); }
      res.cookie('user', JSON.stringify(req.user.user_info), { expires: new Date(Date.now() + 1000 * 60 * hrs)} ); // <--
      return res.send(200); // <--
    });
  })(req, res, next);
}
