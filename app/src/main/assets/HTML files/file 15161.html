There are probably other and much better ways of doing what you really need, but:

if ($(window.width() > "769"){

Should be:

if ($(window).width() > 769){

Full code:

var width = $(window).width();
$(window).resize(function() {
    if (width > 769 && $(window).width() < 769) {
        location.reload();
    }
    else if (width < 769 && $(window).width() > 769) {
        location.reload();
