the function postName doesn't need to have these vars passed to it $scope, $http it only needs the dataToPost the other two vars are already accessible. Your function should just look like this 

function postName (dataToPost) {

  $http.post('/name', dataToPost).
  success(function(data) {
    $scope.error = false;
    $scope.data = data;
  }).
  error(function(data) {
    $scope.error = true;
    $scope.data = data;
  });
}
