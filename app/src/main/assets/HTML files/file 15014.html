Add this in your functions.php:

class List_Append extends Walker_Page
{
    function end_el(&$output, $page, $depth)
    {
        $output .= "</li>";
        $output .= "your_ul_here";
    }
}

Then call like so:

wp_list_pages(array('include' => '4', 'title_li' => '', 'walker' => new List_Append()));
