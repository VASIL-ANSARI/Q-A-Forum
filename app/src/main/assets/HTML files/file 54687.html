Given that the value is only a number without any text you could do that:

var max = 627;

function update_progress() {

  if(parseInt($('#timer').val()) < max) {
      $('#timer').val(parseInt($('#timer').val()) + Math.floor(Math.random() * 50 + 1));

    $('#bar').css('width', parseInt($('#timer').val()) /max * 100 + '%');
  }
    setTimeout(update_progress, 1000);
}

update_progress();

Of course you replace the random stuff with your actual progress. In case there is text aswell, look into substr(); Works aswell with non-inputs (so a <p> for example) in this case you just use .html() instead of .val().
