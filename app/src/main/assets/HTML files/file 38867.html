No, you cannot. Because ParentClass instance is not yet created. You need something like the following.

list.Select(o =>
{ 
   var parent = new ParentClass
    {
        ID = o.ID
    };
    parent.ChildClass = o.Childs
                        .Select(p =>
                           new ChildClass 
                           { 
                               Parent = parent,
                               ID = p.id 
                           })
                         .ToList();
     return parent;
});
