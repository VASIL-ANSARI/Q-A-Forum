In success handler you have JSON object which you can use to populate the table.

...
success: function(data){
   var tbody = $('.table.table-striped > tbody');
   $(data.users).each(function(index, user){
     $('<tr />')
       .append($('<td />').html(user.Fecha))
       .append($('<td />').html(user.Categoria))
       .appendTo(tbody);
   });//each
}//success
