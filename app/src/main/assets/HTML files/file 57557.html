You didn't send the object to your service properly, you should define an object of key/value pairs that contains name and value of your variables.so you should change your ajax request like this:

var value = $("#newval").val(); 
$.ajax({
    type: "Put",
    contentType: "application/json; charset=utf-8",
    url: "/api/updateparameter/" + global_id,
    data: {'value': value},
    success: function () { HidePopUp(); }
});
