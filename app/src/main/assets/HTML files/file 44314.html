This will give you an array of property names, which you could then iterate over (adapted from http://stackoverflow.com/a/328418/4670652):

function propertiesArray(obj) {
   var propList = [];
   for(var propName in obj) {
      if(typeof(obj[propName]) != "undefined") {
         propList.push(propName);
      }
   }
   return propList;
}
