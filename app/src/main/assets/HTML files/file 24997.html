This function already exists within the standard library, see std::set_union, which you can use as follows:

std::vector<int> vec1 = {1, 2, 4}, vec2 = {5, 4, 2}, vecUnion;

std::set_union( vec1.begin(), vec1.end(), vec2.begin(), vec2.end(), std::back_inserter(vecUnion) );

for( auto i : vecUnion )
{
     std::cout << i << ", " std::endl;
}

Which will print out the following:

1, 2, 4, 5,
