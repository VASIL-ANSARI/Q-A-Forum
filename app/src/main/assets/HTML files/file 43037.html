$arr=array('aa','bb','cc');
$str='A_STRING_bb_WITH_SOME_TEXT aa';


function find_which_one(array $arr,$str) {
    $r = preg_match(
        '/((' . implode(')|(', $arr) .'))/',
        $str,
        $matches
    );
    return ($r !== false) ? $matches[0] : $r;
}
