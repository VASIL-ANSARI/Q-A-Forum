Try to loop through all the lis using selector eq() with index wile the length of lis reached :

$(function(){
    setTimeout(function(){

        var slide_count = $('.navigation li').length;
        var cpt=0;

        setInterval(function() 
        {
            if( cpt < slide_count )
            {
                $('.navigation li:eq('+cpt+')').click();

                cpt++;
            }else{
                cpt=0;
            }
        }, 2000);

    }, 1000);

});
