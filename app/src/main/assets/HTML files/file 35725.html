void SomeClass::Insert (int newNumber)
{
    if (m_map.count (newNumber) > 0)
    {
        m_map[newNumber]++;
    }
    else 
    {
        m_map[newNumber] = 1;
    }
}

void SomeClass::Count ()
{
    for (map<int,int>::iterator it=m_map.begin(); it!=m_map.end(); ++it)
    {
        std::cout << "the number: "<< it->first << " apears " << it->second << "times" <<'\n';
    }
}
