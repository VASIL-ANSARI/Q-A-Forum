$(function(){
    function buttonState(){
        var isValid = true;
        $("input").each(function(){
            if($(this).val() == "" ) isValid = false; 
        });
        if (isValid) $('#submit').attr('disabled', '');
        else $('#submit').attr('disabled', 'disabled');
    }
    $('#submit').attr('disabled', 'disabled');
    $('input').on('change keyup', buttonState);
})
