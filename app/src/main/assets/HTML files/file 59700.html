Refer this example:

$('#form_id').validate({
    rules:{
        uname : 
        {
            required: true
        },
        email : 
        {
            required: true,
            email : true
        }
    },
    messages:{
        uname : 
        {
            required: "message"
        },
        email : 
        {
            required: "message",
            email : "message"
        }
    }
});

where form_id is form id of the form for which validation is created. Form must contain two fields with name uname, email.
