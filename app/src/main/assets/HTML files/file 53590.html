You can use an istream_iterator together with the vector constructor taking an iterator pair. Here is an example of a program placing its own source code in such a vector (and then printing it out):

#include <vector>                                                                                                                               
#include <string>
#include <fstream>
#include <iostream>
#include <iterator>


using namespace std;


int main()
{
    ifstream in{"stuff.cpp"};
    vector<string> words{
        istream_iterator<string>{in},
        istream_iterator<string>{}};
    for(const auto &s: words)
        cout << s << endl;
}

(If you use this, make sure to build using c++11 settings.)
