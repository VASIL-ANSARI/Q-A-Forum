Perhaps a UNION will suit you well:

SELECT
    'test' AS `search_term`,
    COUNT(*) AS `term_count`
FROM table
WHERE text LIKE '%test%'
UNION
SELECT
    'test2' AS `search_term`,
    COUNT(*) AS `term_count`
FROM table
WHERE text LIKE '%test2%'
UNION
SELECT
    'test3' AS `search_term`,
    COUNT(*) AS `term_count`
FROM table
WHERE text LIKE '%test3%'
