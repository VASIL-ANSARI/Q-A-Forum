The difference between the working (second) snippet and the first snippet is that the first snippet shadows the mColumns variable. By defining a formal function parameter with the same name, the original variable becomes invisible to code inside the function:

define(["collections/mColumns"],function(mColumns){  // <-- mColumns (#1)
    ...
    return Backbone.View.extend({
        ...
        initialize: function(mColumns){              // <-- mColumns (#2)
            console.log(mColumns);                   // <-- Refers to #2
        }
    })
});
