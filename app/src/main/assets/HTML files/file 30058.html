1) On the client, the variable Meteor.isServer is false, so the if block is not executed, so the HTTP call does not happen and the variable is not initialized.

2) Use a method:

var dayData;

...

if(Meteor.isServer) {
  Meteor.methods({
    getDayData: function() {
      return dayData;
    },
  });
}


if(Meteor.isClient) {
  Meteor.startup(function() {
    Meteor.call('getDayData', function(error, result) {
      dayData = result;
    });
  });  
}
