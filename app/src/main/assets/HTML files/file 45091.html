if you are on localhost you should have to use phpmailer library or swiftmailer, for sending html mail you have to set html headers "Content-type:text/html;charset=UTF-8" for example

<?php
$to = "somebodyelse@example.com";
$message .= "<html><body><a href='http://www.rudraliving.com/brochure_images/$fname'>Download Attachment</a></body></a></html>"."\n\n";>


// Always set content-type when sending HTML email
$headers = "MIME-Version: 1.0" . "\r\n";
$headers .= "Content-type:text/html;charset=UTF-8" . "\r\n";

// More headers
$headers .= 'Cc: myboss@example.com' . "\r\n";

mail($to,$subject,$message,$headers);
?>
