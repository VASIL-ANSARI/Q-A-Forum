var myFadeIn=function(hash){
    hash.w.fadeIn("2000");
};

var myFadeOut=function(hash){
    hash.w.fadeOut("2000",function(){
        hash.o.remove(); 
    });
};

$('.trigger').click(function(){
    $('.jqmWindow').jqm({
        ajax:'people',
        onShow:myFadeIn,
        onHide:myFadeOut
    });
    $('.jqmWindow').jqmShow();
});

$('.jqmWindow').on('click', '.close', function(){
    $('.jqmWindow').jqmHide();
});
