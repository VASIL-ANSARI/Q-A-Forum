Use the map::iterator and the find() function for maps. Once the value is found in the map, the set is searched:

bool found = false;
TFlights::iterator it = flights.find("Madison");
if ( it != flights.end())
{
   // key exists in map.  Now search the set
   if (it->second.find("Boston") != it->second.end())
     found = true;
}
return found;
