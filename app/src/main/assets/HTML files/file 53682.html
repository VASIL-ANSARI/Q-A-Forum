I figured out a way to not use $watch at all, and to update the height once the DOM loads. It was very simple, and I feel silly for not seeing this before. I hope this helps someone before they spend hours trying to do something really complicated like I tried.

angular.module('task2')
.directive('centerElement', function ($timeout) {
    return function (scope, element, attrs) {
        angular.element(document).ready(function () {
            element.css('margin-top', '-' + (element[0].offsetHeight / 2) + 'px');
            element.css('margin-left', '-' + (element[0].offsetWidth / 2) + 'px');
        });
    };
});
