try to use jquery's active and prev:

js :

$(document).ready(function(){
var active = $(".active");
var next = active.next('img');
active.click(function(){
    active.removeAttr('class');
    next.addClass('active');
    });
})

html : 

<ul>
<li>
<img src="1.jpg" class="active"/>
<img src="2.jpg" class=""/>
</li>
</ul>
