this is not working because Local Value has higher precedence that Trigger Value . Dependency Property Precedence

Try it using Converter

xaml

<Window x:Class="Stackoverflow.Window2"
    xmlns="http://schemas.microsoft.com/winfx/2006/xaml/presentation"
    xmlns:x="http://schemas.microsoft.com/winfx/2006/xaml"
    Title="Window2" Height="300" Width="300"
    xmlns:local="clr-namespace:Stackoverflow"
    >
<Window.Resources>
    <local:TextConverter x:Key="textConverter"/>
</Window.Resources>

<Grid>
    <TextBox Text="{Binding Name, Converter=textConverter}"/>
</Grid>

Converter

    public class TextConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, System.Globalization.CultureInfo culture)
    {
        if (value != null && value.ToString() == "-1")
            return "Default";
        return value;
    }

    public object ConvertBack(object value, Type targetType, object parameter, System.Globalization.CultureInfo culture)
    {
        throw new NotImplementedException();
    }
}
