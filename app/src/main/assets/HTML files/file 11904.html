example for firstname and address:

NSError* error;
NSDictionary* json = [NSJSONSerialization 
                  JSONObjectWithData:data 
                      options:kNilOptions 
                      error:&error];

NSString *firstname = [json objectForKey:@"firstname"]; 
NSDictionary *address = [json objectForKey:@"address"];
