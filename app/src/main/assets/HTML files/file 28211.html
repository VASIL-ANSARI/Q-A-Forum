Here is a generic idea of what you want:

        const long BILLION = 1000000000;
        const long MILLION = 1000000;
        const long THOUSAND = 1000;

        long a = 1256766123;
        long b;
        string c = string.Empty;

        if (a >= BILLION)
        {
            b = a / BILLION;
            c += b.ToString() + "B";
            a -= (b * BILLION);
        }

        if (a >= MILLION)
        {
            b = a / MILLION;
            if (c != string.Empty)
                c += " ";
            c += b.ToString() + "M";
            a = a - (b * MILLION);
        }

        if (a >= THOUSAND)
        {
            b = a / THOUSAND;
            if (c != string.Empty)
                c += " ";
            c += b.ToString() + "T";
        }

And variable c will contain your string.
