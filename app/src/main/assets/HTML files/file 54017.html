To insert at the begining of the array use unshift

array.unshift({
    Name: null,
    Age: null,
    Sex: null
});

to remove you can use filter

function remove(names) {
    array = array.filter(function(obj) {
        return names.indexOf(obj.Name) == -1;
    });
}
remove(['Brian', 'Rick']);

or if you want to pass string use:

function remove(names) {
    names = names.split(',');
    array = array.filter(function(obj) {
        return names.indexOf(obj.Name) == -1;
    });
}
remove('Brian,Rick');
