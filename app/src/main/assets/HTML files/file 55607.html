Here is something I tried. This may not be the optimized, but working for me.

<script>

var flg = 0;    

$("tr").find("td").each(function () {

    if ($(this).attr('class') != '' && $(this).attr('class') != undefined) {            

        if ($(this).hasClass('III')) {

            if (flg != 2) {

                $(this).css('background-color', 'red');

                flg = 1;

            }

        } else {

            if (flg == 1) {

                flg = 2;

            }

        }

    }

});

</script>
