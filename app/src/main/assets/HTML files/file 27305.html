public BufferedImage iconToImage(ImageIcon icon) {
  BufferedImage bi = new BufferedImage(
      icon.getIconWidth(),
      icon.getIconHeight(),
      BufferedImage.TYPE_INT_ARGB);
  Graphics g = bi.createGraphics();

  // paint the Icon to the BufferedImage.
  icon.paintIcon(null, g, 0,0);
  g.dispose();

  return bi;
}
