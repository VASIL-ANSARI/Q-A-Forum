      $(document).ready(function() {	

    

		var expanded = false;



		$( ".container ul li" ).click(function(e) {

            var li =$(this);

			if (expanded) {

                li.after($( ".expanded" ));

				$( ".expanded" ).slideUp("slow");

				expanded = false;

			}

			else {

                li.after($( ".expanded" ));

				$( ".expanded" ).slideDown("slow");

				expanded= true;

			}

		});	

	});
