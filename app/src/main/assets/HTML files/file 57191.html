Use OUTER APPLY 

SELECT     
    pev.ID, 
    pev.JobNumber, 
    pev.VersionNumber, 
    pev.DateCreated, 
    pev.CreatedBy,     
    pe.LastModifiedDate,
    pe.ModifiedBy
FROM         
    dbo.t_ProjectEquipmentVersions  pev
OUTER APPLY (
    SELECT TOP(1) pe.ModifiedDate AS LastModifiedDate
        ,pe.ModifiedBy 
    FROM dbo.t_ProjectEquipment pe
    WHERE pe.ModifiedVersion = pev.VersionNumber 
    ORDER BY ModifiedBy DESC
) pe
