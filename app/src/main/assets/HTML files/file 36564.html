var counter = <?php echo $currentGames; ?>,
counterTimer,
totalGames = <?php echo $totalGames; ?>;

counterTimer = setInterval(reCounter(), 5000);

function reCounter() {
    window.title = counter + '/' + totalGames + ' Auto Battles Remaining';

    if (counter <= 0) {
        clearInterval(counterTimer);
    }
    counter--;
}
