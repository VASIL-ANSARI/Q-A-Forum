public static void playSound(Context context, int soundID) {
    if(soundPool == null || soundPoolMap == null){
        initSounds(context);
    }
    float volume = (float) 1.0;
    soundPool.play((Integer) soundPoolMap.get(soundID), volume, volume, 1, 0, 1f);
}
