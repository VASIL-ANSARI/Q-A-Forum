I had the same problem before, then i putted this into my controller:

if ($request->isPost()) {
        $post = array_merge_recursive(
            $request->getPost()->toArray(),
            $request->getFiles()->toArray()
        );
        // To get the required error message
        if (!$post['picture']['tmp_name']) {
            $post['picture'] = null;
        }
        $form->setData($post);
 }
