Assuming your input contain only '}', '{' and '(', ')' characters.

replace by below code (assuming push() and pop() do work fine)...

public void analyze(String input, int length, int j){
        if (j >= length) return;
        if(input.charAt(j) == '{' || input.charAt(j) == '(')
            push(input.charAt(j));
        else if(input.charAt(j) == '}' || input.charAt(j) == ')')
            pop();
        analyze (input, length, ++j);
    }
