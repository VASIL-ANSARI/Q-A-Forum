I don't think you need to depend on 'text' directly. 

define([
  'angular', 
  'services', 
  'text!ang/templates/registry.html'
], function(angular, services, template) {
  //...
  angular.module('myApp.directives', ['myApp.services'])
         .directive('superd', function() {
           return {
             templateUrl: template,
             //...
           };
         });

And in your require config:

require.config({
  //...
  paths: {
    // path example if you are using bower
    'angular': '/bower_components/angular/angular',
    'text': '/bower_components/requirejs-text/text'
  },

  shim: {
    'angular': {
      exports: 'angular'
    }
  }
});
