You need something like this.

$("select").change(function () {
    var counter = 0;
    var arr = $(this).map(function(){ return $(this).val(); }).get().map(function(x){
        return "color" + (++counter) + "=" + x;
    });
    $('img').attr('src', function (_, src) {
        return src.replace(/\?.*/,"?") + arr.join("&");
    });
});
