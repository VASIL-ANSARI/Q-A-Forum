Closures work in the scope that the functions are created in. Not the scope they are executed in. For example, the following should work:

server.get('/:user_id', function(req, res, next) {
  var fn1 = function(callback){
      console.log(req.params.user_id);
      callback(null, req.params.user_id);
  };

  var fn2 = function(user_id, callback){
      console.log(user_id);
      callback();
  }

  async.waterfall([fn1, fn2],
    function(err, result){
      next();
    });
});
