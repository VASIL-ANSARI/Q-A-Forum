What you are looking for is something like this:

#include <stdlib.h>

#if defined unix        ||                                                     \
    defined __unix      ||                                                     \
    defined __unix__    ||                                                     \
    defined __linux__   ||                                                     \
    defined __FreeBSD__ ||                                                     \
    defined __CYGWIN__  ||                                                     \
    (defined __APPLE__ && defined __MACH__)
    static const char TMP_DIR[] = "~/.temp/";
#elif defined WIN32   ||                                                       \
      defined _WIN32  ||                                                       \
      defined __WIN32
    static const char TMP_DIR[] = "C:\\TEMP\\";
#else
    #error "Platform not supported"
#endif

int
main(void)
{
    printf("%s\n", TMP_DIR);
    return EXIT_SUCCESS;
}
