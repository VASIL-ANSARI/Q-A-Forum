jQuery code for a colorbox that automatically resizes based on jQueryui accordion content:

$(function () {
    $("#accordion").accordion({
        heightStyle: 'content',
        collapsible: true,
        active: false,
        activate: function( event, ui ) {
            $.colorbox.resize();
        }
    });
});

$(".inline").colorbox({
    inline: true,
    innerWidth: '90%',
    scrolling: false      
});
