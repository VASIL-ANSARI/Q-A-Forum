Just wrapping it on a loop and using .find():

$(".container").each(function(){
   boxes = $(this).find('.heightHack');
   maxHeight = Math.max.apply(
     Math, boxes.map(function() {
       return $(this).height();
   }).get());
   boxes.height(maxHeight);
});
