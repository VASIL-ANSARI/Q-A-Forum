The following code will build 3 strings (one for each TextBox) and later assign them to their corresponding Text properties. It uses StringBuilder to optimize concatenation. AppendLine is useful to keep spaced apart the different concatenated values: .Replace(Environment.NewLine, ", ") will later replace line breaks with ", "; if you don't need to space apart the values, simply remove .Replace(Environment.NewLine, ", ") and replace AppendLine with Append.

protected void Button1_Click(object sender, EventArgs e)
{
    var sbText1 = new StringBuilder();
    var sbText2 = new StringBuilder();
    var sbText3 = new StringBuilder();

    var returnedData = Getdata();
    foreach (var dataclass0 in returnedData)
    {
        sbText1.AppendLine(dataclass0.item1);
        sbText2.AppendLine(dataclass0.item2);
        sbText3.AppendLine(dataclass0.item3);
    }

    TextBox1.Text = sbText1.ToString().Replace(Environment.NewLine, ", ");
    TextBox2.Text = sbText2.ToString().Replace(Environment.NewLine, ", ");
    TextBox3.Text = sbText3.ToString().Replace(Environment.NewLine, ", ");
}
