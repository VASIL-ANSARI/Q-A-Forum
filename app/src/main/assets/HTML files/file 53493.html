  dfbmin = parseInt(splitdfbrun[0]) + parseInt(splitdfbpen[0])
  dfbmin = dfbmin % 60;

  dfbsec = parseInt(splitdfbrun[1]) + parseInt(splitdfbpen[1])

  // Update Minutes
  if (dfbsec >= 60) {
