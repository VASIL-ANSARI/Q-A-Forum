Populate the table with the AJAX content

   $("#cmdUnreadID").click(function(){
       $.ajax({
                type: "GET", 
                url:"orderBy.php", 
                data: "unreadCtrl=1", 
                success:function(result){
                  $("#tableInbox").hide(500,function(){
                    $("#tableInbox").html(result).show(); // <-- Notice html() call here, to populate the table
                  });
                }
        });
   });
