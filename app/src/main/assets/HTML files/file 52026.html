$records = Sendqueue::select()
            ->where('userID', Session::get('uid'))
            ->where('campaign', $id)
            ->where('status', 'pending')
            ->get();

foreach ($records as $record) {
        DB::table('sendqueue')
            ->where('ID', $record->ID)
            ->update(['status' => 'stopped']);
}
