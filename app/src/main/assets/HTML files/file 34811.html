using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace ConsoleApplication3
{

    {

        Console.Write("what is your age?");

            if (age >= 21)
            {
                Console.WriteLine("congrats, you're you can get drunk!");

            }
            else if (age < 21)
            {
                Console.WriteLine("Sorrrrrryyyyyyy =(");

        else
