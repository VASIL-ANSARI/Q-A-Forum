Imagine you have two rows #row1 and #row2. Then you can traversing through original collection (which in #collection block) this way:

var $row1 = $('#row1');
var $row2 = $('#row2');
$('#collection div').each(function(i) {
    if (i % 2) {
        $(this).appendTo($row2)
    } else {
        $(this).appendTo($row1)
    }
});
