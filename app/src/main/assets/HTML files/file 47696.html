Jquery Ajax bind with click event

<script>
        $("#loadContent").on("click",function(event){
            event.preventDefault();
            $.ajax({
                url: $(this).attr("href"),
                type: 'GET',
                data: {"anydata":"anydatavalue"},
                success : function(reponse){
                    $("#center").html(reponse);
                }
            }); 
        });

    </script>
