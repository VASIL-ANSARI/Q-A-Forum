One of the ways to overcome this is to create a function and call in within the controller and since the controller is attached to the view it will get updated every time you enter the view . Something like this 

$scope.onLoad = function() {
    $scope.nodeId = $stateParams.studynodeRef;
    Studies.fetchCollection($scope.nodeId).then(function(data){
        $scope.y = angular.fromJson(data.list);
        $scope.collections = angular.fromJson($scope.y[0].collections);
        console.log($scope.collections);
    })
}
$scope.onLoad();
