I only had to push the object into an array and draw the bars from this array. 

var datatemp = [];

datatemp.push(data)

// draw the bars
        svg.selectAll(".bar")
        .data(datatemp)
        .enter()    
            .append("rect")
            .attr("class", "bar")
            .attr("id", function(d) {return d.CountryName })
            .attr ({
                "x": function(d) {return xScale(d.CountryName);},
                "y": function(d) {return yScale(d.Variable);},
                "width": xScale.rangeBand(),
                "height": function(d) { return height - yScale(d.Variable);}
            })
            .on('mouseover', tip.show)
            .on('mouseout', tip.hide);
