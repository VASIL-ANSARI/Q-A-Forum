You could use a decorator that prevents your function from being run more than once:

from functools import wraps

def run_once(function):
    @wraps(function)
    def wrapper(*args):
        if not wrapper.has_run:
            wrapper.has_run = True

            return function()

    wrapper.has_run = False

    return wrapper

Like so:

@run_once
def signal_handler():
    print 'Exit'
