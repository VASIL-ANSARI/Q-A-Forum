You can do this by creating a subclass of frozenset and overriding its __repr__ method:

class MyFrozenSet(frozenset):
    def __repr__(self):
        return '([{}])'.format(', '.join(map(repr, self)))
...     
>>> lst = [['yes'], ['red', 'blue']]
>>> [MyFrozenSet(x) for x in lst]
[(['yes']), (['blue', 'red'])]
