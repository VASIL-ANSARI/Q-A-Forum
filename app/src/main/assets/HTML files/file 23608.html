I am not trying to fix your code but why you don't use this ajax format to handle success/failure ?

$.ajax({
    type: "POST",
    url: templateURL + "/post.php",
    data: dataString,
    dataType: "json"
}).done(function () {
    //success
    $.fancybox("success", {
        // options
    });
}).fail(function () {
    //error
    $.fancybox("failure", {
        // options
    });
}).always(function () {
    // optional after ajax is completed
    $.fancybox("else", {
        // options
    });
});
