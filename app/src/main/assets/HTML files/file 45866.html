Try:

var v, i;
var a = {};
a.x = "randomvalue";
for (x in a) {
  if a[x] === "randomvalue" {
    v = x;
    i = "randomvalue";
  }
}

Then v contains the object key and i contains the object value (although you don't really need it).

Or, if you know the value index:

var obj = { first: 'someVal' };
obj[Object.keys(obj)[0]]; //returns 'someVal'
