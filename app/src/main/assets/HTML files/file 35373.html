To check whether a string consists of 4 digits you can write a if statement as,

int a,b,c,d,id;
if(isdigit(hyp[0])&&isdigit(hyp[2])&&isdigit(hyp[4])&&isdigit(hyp[6]))
   {
       sscanf(hyp,"%d %d %d %d",&a,&b,&c,&d);
       id=a*1000+b*100+c*10+d;
   }
