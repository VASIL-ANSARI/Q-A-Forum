You want to retrieve headers from a request without connecting? I'm not sure what you trying to do, but have you look into node's core module 'http'?

Example:

var require('http');

http.request({
    host   :   'localhost',
    port   :   8080,
    method :   'GET',
    path   :   '/endpoint'
}, function(res)
{
   res.on('end', function()
   {
       console.log('My headers: %O', res.headers);
       res.end();
   }
});
