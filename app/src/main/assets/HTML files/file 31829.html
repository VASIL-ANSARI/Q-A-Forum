Use is_a?:

def your_method(param)
    if param.is_a? String
        # it's a string
    else
        # not a string
    end
end

Or, if you have lots of classes to check, a case statement:

def your_method(param)
    case param
    when String
        # it's a string
    when Array
        # it's an array
    when Rational
        # it's a rational
    # ...
    else
        # it's not any of these classes
    end
end
