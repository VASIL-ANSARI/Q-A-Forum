If you don't want to retain a copy of REPORT.xls then something like below. But I would suggest you write a separate SQL Loader script for this. 

sqlplus -S <<ENDOFSQL  || eval 'echo "Error occurred during sqlplus command line $LINENO" 1>&2; exit 1'
$USER/$PW@//$IP_PORT/$DB
    SET PAGESIZE 50000
    set feedback off
    SET MARKUP HTML ON SPOOL ON
    SET NUM 24
    SPOOL report.sql

    select 'insert into SENT values('||column1||','||column2...||');' 
    from SOMETABLE A 
    WHERE not exists (select 1 
                        from SENT_TABLE B 
                        where   A.ID = B.ID);

    SPOOL OFF;
    @report.sql 
    COMMIT;
ENDOFSQL
