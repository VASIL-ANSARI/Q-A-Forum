Change the onSubmit event handler to a onClick handler on the submit button. The onSubmit event is fired when the data is posted to the server and is thus too late to prevent a reload.

 $('#submit_button').on('click',function(e){

        e.preventDefault();

        var that = $('#contact_form');

        var url = that.attr('action');

        var action = that.attr('method');

        $.post(url,action,that.serializeArray())
        .done(function (data,text,jqhr) {

                console.log('data');
            })
        .fail(function(jqxhr){
            alert(jqxhr.responseText)
        })

        .always(function () {
            that.find('button').text('Send Message');
        })
    });

});
