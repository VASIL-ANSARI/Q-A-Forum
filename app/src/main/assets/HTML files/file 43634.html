*most (from my experience) JSON serializers can only handle Dictionary as long as they are dealing with primitive (and string) key/values, in other words no you'd have to either build your own handler or return 2 seperate lists.

This would work:

var Cache = new Cache()
{
    Type = "Cache",
    Id = "Cache",
    ElementKeys = new List<string> {
        {"Parking", "House" };
    ElementValues = new List<JSONNode>{
        new Parking(){
        Type = "Parking",
        Id = "Parking",

        Cars = new List<Car>(){
            new Car(){
            Type="MyType1",
            Id = "3",
            Back = null},
            new Car(){
            Type="MyType2",
            Id = "3",
            Back = null}}
            },
        new House(){
        Type="House",
        Id="House",
        Number=156}

    }
 };

You'd have to change your Cache class like this:

[Serializable]
public class Cache : JSONNode
{
    public List<string> ElementKeys { get; set; }
    public List<JSONNode> ElementValues { get; set; }
}
