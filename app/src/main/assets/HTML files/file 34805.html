Try this and you will see the output in each step... the final is 1

var v = [2, 4, 5];
var out = 0;
for (var i = 0; i < v.length; i++) {
    if(v[i] % 2 === 0){ out++; } 
    else { out--; }
    console.log(out);
}
// final value again
console.log(out);
// result
// 1
// 2
// 1
// 1

// PS. you can also use a ternary (just for info)
out += v[i] % 2 ? -1 : 1;
