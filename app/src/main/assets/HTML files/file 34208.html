If you're using pure JS then just use a for statement and pull from an array:

var strings = ['Hotel', 'Room'];
for(var i = 0; i < strings.length; i++){
    $("a[rel='" + strings[i] + "']").colorbox({
        maxWidth: "90%",
        maxHeight: "90%",
        opacity: ".5"
    });
}
