If you are parsing CSV I would recommend that you use a tool like CsvHelper. We have been using it for a very similar task. It has all sorts of awesome configuration. 

var csvConfiguration = new CsvConfiguration
{
    Quote = '"',
    Delimiter = ",",
    TrimFields = true,
    TrimHeaders = true,
    QuoteNoFields = false,
    QuoteAllFields = false,
    HasHeaderRecord = true,
    IsHeaderCaseSensitive = false,
    WillThrowOnMissingField = false
};
