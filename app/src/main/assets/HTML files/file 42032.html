You have to display elements as inline-block and set vertical-align: middle, text-align: center.

#container {

  display: inline-block;

  height: 80px;

  width: 100%;

}

#one,

#two,

#three {

  display: inline-block;

  text-align: center;

  vertical-align: middle;

}

#one {

  width: 20%;

  font-size: 90px;

}

#two {

  width: 20%;

  font-size: 50px;

}

#three {

  width: 20%;

  font-size: 60px;

}
