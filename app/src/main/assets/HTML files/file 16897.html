SELECT *
FROM   MY_TABLE f 
WHERE  NOT EXISTS (
                    SELECT 1 
                    FROM MY_TABLE t 
                    WHERE f.`from` = t.`to` 
                    AND f.`to` = t.`from` 
                    AND f.id > t.id)
