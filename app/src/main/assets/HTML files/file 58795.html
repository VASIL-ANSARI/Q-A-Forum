Elasticsearch allows you to sort according to a field, I'm not familiar with Rails but I guess you should try this:

@response = Elasticsearch::Model.search query:     {
  "indices": {
    "indices": [
      "shares",
      "rents"
    ],
    "sort": {
       "create_at": {
          "order": "desc"
       }
    },
    "query": {
      "multi_match": {
        "query": "Apartament",
        "fields": [ "tipe"]
      }
    }
  }
}
