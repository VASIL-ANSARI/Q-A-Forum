Your array values do not have directory entries, thus the children are loading from the wrong directory. Either change the array entry to include full paths, or include a base directory attribute for the image that you read with the HTML5 data attribute. I've included a working example of the later - attempting to leave your original script as intact as possible.

It's bad practice to constantly re-lookup a selector, so I did make one small optimization and store the $('#picture') selector in a variable. Technically, you should try to limit yourself to one var declaration in the method as well, although it will work as you have it.

$(document).ready(function() {  

    var imageName=["Daisy.jpg","Diamond.jpg","Heart.jpg","Unique.jpg"];

    var imageTitle=["Daisy","Diamond","Heart","Unique"]

    var counter = 0;

    var $pic = $('#picture');

    function go() {

            $pic.fadeOut(3000, function(){

                $pic.attr("src", $pic.data('basedir') + '/' + imageName[counter]);

                $("#imageCaption").text(imageTitle[counter]);

                counter++;

                if (counter >= imageName.length) {counter = 0;}

                $pic.fadeIn(900).delay(7000);

                go();

            });

        }

    go();

    $pic.click(go);

});//end of document ready
