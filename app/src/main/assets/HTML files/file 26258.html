Try to update only graphic using element.attr(), see: http://jsfiddle.net/rh5fz92q/

    plotOptions: {
        pie: {
            point: {
                events: {
                    mouseOver: function(e){
                        this.defaultR = this.graphic.r;
                        this.graphic.attr({
                            r: 200
                        });
                    },
                    mouseOut: function(e){
                        this.graphic.attr({
                            r: this.defaultR
                        });
                    }
                }
            }
        }
    },
