You can try this query

SET @level = 0;
SET @group = '';
SELECT 
    item_id,
    category_id
FROM (
    SELECT 
        item_id,
        category_id,
        @level := IF(@group = category_id, @level+1, 1) AS level, 
        @group := category_id as EGroup 
    FROM item
    ORDER BY category_id 
) rs
WHERE level < 5
