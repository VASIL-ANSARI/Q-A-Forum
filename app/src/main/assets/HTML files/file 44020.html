On a quick glance I can see a couple of errors in the JS. Javascript variables are created with a var syntax like var size_li = $("#myList li").size(); or var x = 3. Once these are created we can assign them values directly like x = 4. 

$(document).ready(function () {
    var size_li = $(".carousel-row").size();
    var x=3;
    $('.carousel-row:lt('+x+')').show();
    $('#loadMore').click(function () {
        x= (x+5 <= size_li) ? x+5 : size_li;
        $('.carousel-row:lt('+x+')').show();
    });
    $('#showLess').click(function () {
        x=(x-5<0) ? 3 : x-5;
        $('.carousel-row').not(':lt('+x+')').hide();
    });
});

The #showMore and #showLess buttons need to be added to the html to trigger the click functions also, but we are now targetting .carousel-row to check how many rows exist.
