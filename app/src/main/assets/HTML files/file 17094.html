Iterate over each Row instead:

var $divItems = $( "<div/>", {
    "class" : "items"
});

$(data).find("Row").each(function(i, item) {
    var heures = $(item).attr("Dim1");
    var nbr = $(item).attr("Metric1");
    $divItems.clone()
        .html('<a href="' + heures + '">' + nbr + '</a>')
        .appendTo("#xml");
});
