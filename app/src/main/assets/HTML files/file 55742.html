var $flippers = $(".flip-container"),

    qtFlippers = $flippers.length;

var isFlipped = false;

var randomItem = -1

setInterval(function () {

    

    if (isFlipped && randomItem !== -1) {

    	$flippers.eq(Math.floor(randomItem)).toggleClass('hover');

    	isFlipped = false;

      randomItem = -1;

      return;

    }

    

    if (!isFlipped) {

      randomItem = Math.random()*qtFlippers;

    	$flippers.eq(Math.floor(randomItem)).toggleClass('hover');

    	isFlipped = true;

    }

}, 1000);
