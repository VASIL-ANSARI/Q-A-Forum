If you want an array, why don't you declare it as an array?

struct A {
    const char *str;
    int n;
};

struct A b[3] = {
    {
        "foo", 1
    },
    {
        "bar", 2
    },
    {
        "baz", 3
    }
};

If you want a global pointer, then use a global pointer:

struct A *b;

int main()
{
    b = malloc(sizeof(*b) * 10);

    // do stuff

    free(b);
    return 0;
}
