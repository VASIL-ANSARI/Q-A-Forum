   private void ImageToPixcels()
    {
        Bitmap img = (Bitmap)Image.FromFile(@"D:\sample.jpg");
        Color[,] pixels = new Color[img.Width, img.Height];
        for (int x = 0; x < img.Width; x++)
        {
            for (int y = 0; y < img.Height; y++)
            {
                pixels[x, y] = img.GetPixel(x, y);
            }
        }
    }
