You could make a subquery

update Email
   set status        = 'I',
       preferred_ind = 'N'
 where Email.pid in (select E.pid
                       from Email E
                      inner join Person P on P.pid = E.pid
                      inner join Temp T on T.id = P.pid
                      where P.change_ind is null
                        and E.status = 'A'
                        and E.email_addr <> T.email_addr);
