SELECT DISTINCT cte.staff_id 
FROM (
       SELECT st.staff_id, ROW_NUMBER(PARTITION BY st.staff_id ORDER BY s.server_id) AS cnt
       FROM Staff st
       INNER JOIN user_staff u_s ON u_s.staff_id = st.staff_id
       INNER JOIN user u ON u.ssh_user_id = u_s.ssh_user_id
       INNER JOIN server s ON s.server_id = u.ssh_user_server_id
     ) cte
WHERE cte.cnt < ( 
                  SELECT COUNT(s2.server_id)
                  FROM server s2
                )
