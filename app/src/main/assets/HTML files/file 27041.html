You can use jQuery's :not selector. Also, I have switched your click function to on event handler, which is a bit more flexible to work with than the shorthand click. Personal preference there.

jQuery

$('a:not("#flip")').on('click', function(event) {
  event.preventDefault();
  // newLocation = this.href;
  // $('#monster').fadeOut('slow', newpage);
  alert('Only link tags without the #flip id should show this');
});

var newpage = function() {
  window.location = newLocation;
}
