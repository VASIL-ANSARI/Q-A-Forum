I hope this will help you. This code to add border for only in bottom side of view in obj c. 

CALayer *border = [CALayer layer];
border.borderColor = [UIColor whiteColor].CGColor;
CGRect myVFrame = myView.frame;
border.frame = CGRectMake(0, myVFrame.size.height - 2, myVFrame.size.width, myVFrame.size.height);
border.borderWidth = 2;
[myView.layer addSublayer:border];
myView.layer.masksToBounds = YES;
