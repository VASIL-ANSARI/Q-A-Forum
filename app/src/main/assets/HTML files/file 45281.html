Get the th elements by text, locate the user field values using following-sibling and construct a user object:

var user = {
    Title: row.element("//th[. = 'Title:']/following-sibling::td").getText(),
    Email: row.element("//th[. = 'Email:']/following-sibling::td").getText()
};

Example usage:

expect(user).toEqual(TestUser);

Note that user is an object where field values are promises, but, since expect() is patched to resolve promises implicitly, we are good to go - it should make the assertion once all promises are resolved.

If you want to abstract out the field names and make it reusable, use map():

var table = $("table.account-details");
var user = {};

table.all(".//th").map(function (elm) {
    var key = elm.getText(),
        value = elm.element(by.xpath("following-sibling::td")).getText();

    user[key] = value;
});

expect(user).toEqual(TestUser);
