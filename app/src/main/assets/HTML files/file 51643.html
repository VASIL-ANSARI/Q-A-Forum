You will have to wait for the editor to load in order to insert content into it. document.ready won't work in this case. Use:

  tinymce.init({
    selector: "textarea",
    directionality: 'rtl',
    plugins: [
    "advlist autolink lists link image charmap print preview anchor",
    "searchreplace visualblocks code fullscreen",
    "insertdatetime media table contextmenu paste",
    "textcolor"
],
   setup : function(ed) {
      ed.on('init', function(evt) {

      // get your content
      // and insert it into the editor
      ed.setContent(myData);


      });
    },
    toolbar: "insertfile undo redo  | bold italic | alignleft aligncenter alignright alignjustify | bullist numlist outdent indent  | fontselect |  fontsizeselect |  forecolor backcolor"


});
