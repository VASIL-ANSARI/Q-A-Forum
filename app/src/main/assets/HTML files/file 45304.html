You can try this way to get all of the branches when "null" is selected.

select distinct
bta.stn_stn_id
from ody.bil_tas_areas bta
where 
bta.stpr_id = 
(
    select uda.area_id
    from ody.usr_dfnd_areas uda
    where uda.aty_area_typ = 'tas_a_stpr'
    and (uda.AREA_DESC = ? or 'null' = ?)
)
order by 1
