Just put them in an array:

DropDownList[] ddls = new DropDownList[]{
    DropDownList1, DropDownList2,
    DropDownList3, DropDownList4,

};

And loop through it:

foreach (DropDownList ddl in ddls)
{
    ddl.DataSource = ds;
    ddl.DataTextField = "name";
    ddl.DataValueField = "id";
    ddl.DataBind();
}
