When you have all the new rows you need in other_table (just OutTime missing), you can use the following stored procedure to update the missing values row by row. They should exist on every second row as well in the old table.

DELIMITER $$
DROP PROCEDURE IF EXISTS `update_other` $$
CREATE PROCEDURE `update_other` ()
BEGIN
    DECLARE row_counter INT;
    DECLARE what_limit INT;
    DECLARE stop_row INT;
    SET row_counter = 1;
    SET what_limit = 1;
    SELECT COUNT(*) FROM other_table INTO stop_row;

    WHILE row_counter <= stop_row DO
       UPDATE other_table SET Outtime = (SELECT Time FROM process LIMIT what_limit,1) WHERE unique_id = row_counter; 
       SET what_limit = what_limit + 2;
       SET row_counter = row_counter + 1;
    END WHILE;
END $$
