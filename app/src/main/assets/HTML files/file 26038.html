I found the problem after I looked into the generated *.cxx. The SWIG docu IS wrong in two points. Below is the code which is working. Maybe someone wondering about the same issue is helped with it. 

  %{
  #include <AN2DArray.h>
  %}

  %inline %{
      struct Grid2dRow {
          ANN::F2DArray *g;   // Grid
          int    y;      // Row number

          // These functions are used by Python to access sequence types (lists, tuples, ...)
          float __getitem__(int x) {
              return g->GetValue(x, y);
          }

          void __setitem__(int x, float val) {
              g->SetValue(x, y, val);
          }
      };
  %}

  %include <AN2DArray.h>  

  %addmethods ANN::F2DArray {
      Grid2dRow __getitem__(int y) {
          Grid2dRow r;
          r.g = self;
          r.y = y;
          return r;
      }
  };
