You can try this:

$(function () {
$('img').each(function () {
    $(this).data('original', this.src)
}).mouseenter(function () {
    $(this)
        .attr('src', $(this).data('hover'))
        .animate({
        marginTop: 0,
        marginRight: 0,
        marginLeft: 0,
        borderWidth: 10
    }, 'fast')
}).mouseleave(function () {
    $(this)
        .attr('src', $(this).data('original'))
    .animate({
        marginTop: 20,
        marginRight: 10,
        marginLeft: 10,
        borderWidth: 0
    }, 'fast')
})
})
