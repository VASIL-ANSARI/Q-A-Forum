public function getData($table)
{
    $dataArray = array();
    $t = ($table === 'employers') ? 'employers' : 'members';
    $query = "SELECT * FROM $t";
    $stmt = $this->link->prepare($query);
    if($stmt->execute()) {
        $result = $stmt->get_result();
        while($row = $result->fetch_assoc()) {
            $row = array_map('stripslashes', $row);
            $dataArray[] = $row;
        }
    }

    return $dataArray;
}
