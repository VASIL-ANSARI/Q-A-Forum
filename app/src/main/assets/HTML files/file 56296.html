the problem is by definition of Drop Zone. is when you want to add files from clipboard and pass the file to drop zone you have to initialize DropZone. is in dynamic way then every thing will work fine

oh don't forget to set auto discover false :) 

Dropzone.autoDiscover = false;

var myDropZone = new Dropzone("div#dropzoneForm", {url:''});

FileReaderJS.setupClipboard(document.body, {
    accept: {
        'image/*': 'DataURL'
    },
    on: {
        load: function (e, cfile) {           
            $(Dropzone).addfile(cfile);
        }
    }
});
