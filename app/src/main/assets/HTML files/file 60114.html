SELECT
  "ticks"."ts" AS "timestamp"
FROM
  generate_series('2016-01-01'::timestamp, '2016-10-10'::timestamp, '1 month'::interval) AS ticks(ts)
LEFT OUTER JOIN
  (
    SELECT *
    FROM "metrics_raw"
    WHERE "metrics_raw"."type" = 'TRA'
  ) as "metrics"
ON
  "ticks"."ts" = date_trunc('months', "metrics"."timestamp")
GROUP BY "ticks"."ts"
ORDER BY "ticks"."ts"
