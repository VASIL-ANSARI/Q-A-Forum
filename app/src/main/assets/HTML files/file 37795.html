$("#main").droppable({
    tolerance: "pointer",
    drop: function(event, ui){
        var ud = ui.draggable;
        if( ud.draggable( "option", "helper" ) == "clone" ){
            ud.clone().removeClass("display").addClass("area")
            .appendTo("#main")
                .css({
                    position: 'absolute',
                    left: ui.offset.left,
                    top: ui.offset.top
                })
                .resizable()
                .draggable({cursor: "move", containment: '#main'})
                //editDataTypes();
        }else{
        ud.css({
                position: 'absolute',
                left: ui.offset.left,
                top: ui.offset.top
            })
        }
    },
    out: function(event, ui){
        ui.draggable.remove();
    }
});
