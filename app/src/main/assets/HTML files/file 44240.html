$(function () {

$(".one, .two, .three").click(function (event) {
    event.preventDefault();
    $("#" + $(this).attr("class")).fadeIn(function(){
        $(this).siblings('.popup').hide();        
    });
    return false;
});


$(".close").click(function (event) {
    event.preventDefault();
    $(".popup").fadeOut();
});

});
