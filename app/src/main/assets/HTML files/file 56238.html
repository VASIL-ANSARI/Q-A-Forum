It works well with me bro ! I think that's a problem with your compiler. I'm using Gulp
This is my gulpfile.js You can install Gulp and Gulp-sass
Use CMD go to your folder contains gulpfile then run : gulp watch
Everytime you change your scss file it will compile css file for you 

* file: gulpfile.js */

var gulp   = require('gulp'),
    sass   = require('gulp-sass');

gulp.task('build-css', function() {
  return gulp.src('css/*.scss')
    .pipe(sass())
    .pipe(gulp.dest('css'));
});

/* updated watch task to include sass */
gulp.task('watch', function() {
  gulp.watch(['css/*.scss'], ['build-css']);
});
