As stated in the official documentation the callback function accepts index and value: function( index, value ); so in your case i is the index.

This should work: 

$.ajax({
url: url,
success: function(data){
  var cards = data.results;
  console.log(cards);
  $.each(cards, function(i, card) {
    if(i <=  1)
        $("#cards").append('<li> ' + card.category + ' </li>');
    else
        return false; //THis is the correct way to break a jquery each loop
  });
