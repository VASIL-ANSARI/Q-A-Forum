Remove the function from the right of the countdown (demo)

var jquery4u = {
    init: function() {
        jquery4u.countdown.show();
    },
    countdown: {
        show: function() {
            console.log('show');
        },
        hide: function() {
            console.log('hide');
        }
    }
}
jquery4u.init();
