I had the same problem. I had to shange jeditable code to resolve it.

First, I retorned the list from server as a List of SelectListItem (that have this properties: Value, Text, Selected):

listOfEentities.Select(c => new SelectListItem { Text = c.Name, Value = c.Id.ToString() });

Then, change jeditable code (in the part that defines the type select, change the element content, to look like this):

select: {
           element : function(settings, original) {
                var select = $('<select />');
                $(this).append(select);
                return(select);
            },
            content : function(data, settings, original) {
                /* If it is string assume it is json. */
                if (String == data.constructor) {      
                    eval ('var json = ' + data);
                } else {
                /* Otherwise assume it is a hash already. */
                    var json = data;
                }
                var selected = '';
                for (var i in json) {
                    var item = json[i];
                    if (item.Selected)
                        selected = item.Value;
                    var option = $('<option />').val(item.Value).append(item.Text);
                    $('select', this).append(option);    
                }                    
                /* Loop option again to set selected. IE needed this... */ 
                if (selected != "") {
                    $('select', this).children().each(function () {
                        if ($(this).val() == selected ||
                            $(this).text() == $.trim(original.revert)) {
                            $(this).attr('selected', 'selected');
                        }
                    });
                }
            }
        }
