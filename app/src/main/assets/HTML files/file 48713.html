Here is a solution using a double-ended queue. As required, it modifies the list in place, neither uses return nor uses chunks of the list.

from collections import deque

def rotate(lst, x):
    d = deque(lst)
    d.rotate(x)
    lst[:] = d

num = [1,2,3,4,5,6]
rotate(num,3)
print(num)
rotate(num,-3)
print(num)

produces

[4, 5, 6, 1, 2, 3]
[1, 2, 3, 4, 5, 6]
