Use usort() and define a custom sorting function:

usort($array, function($a, $b)
{
    if ($a['apn'] % 2 == $b['apn'] % 2) {
        if ($a['apn'] == $b['apn']) {
            return 0;
        }
        return ($a['apn'] < $b['apn']) ? -1 : 1;
    } 

    return ($a['apn'] % 2 == 0) ? 1 : -1;
});
