from docutils.parsers.rst import tableparser
from docutils import statemachine

parser = tableparser.GridTableParser()
block = statemachine.StringList([
    '+---+---+',
    '| a | b |',
    '| c | d |',
    '+---+---+',
])
print parser.parse(block)
