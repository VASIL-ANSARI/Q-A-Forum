Use a HashMap that maps an action name to its properties: 

interface ActionDetails{
    net:boolean;
    netd:boolean;
}

class Foo{
    // This thing 
    actions:{[name:string]:ActionDetails} = {};

    networkStart = (action: string = null) => {
        this.actions[action].net = true;
        this.actions[action].netd = true;
    }

    networkStop = (action: string = null) => {
        this.actions[action].net = false;
        this.actions[action].netd = false;
    }
}

the "null" will be a default action of sorts.
