ListItem l = list.Items.FindByValue((String)e.Value);
if (null != l)
{
    CssStyleCollection csc = AnswerItemStyle.GetStyleAttributes(list);
    foreach (String key in csc.Keys)
    {
        l.Attributes.CssStyle.Add(key, csc[key]);
    }
