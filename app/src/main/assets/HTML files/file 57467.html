The column.render() / columnDefs.render() methods is for controlling content of cells, not their visual layout. I would suggest the createdRow() callback instead. This is called after a row is inserted, and therefore you can work on the DOM cells directly. Here is an example you can adapt, it colors the cell green if the value is 1 :

var table = $('#example').DataTable({
  createdRow: function( row, data, dataIndex ) {
    var row = this.api().row(dataIndex)
    if (row.data()[0] == '1') {
      row.nodes().to$().find('td:eq(0)').css('background-color', 'green')
    }
  }     
})
