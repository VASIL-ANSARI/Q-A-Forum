You can use reduce() and one forEach() like this.

var arr = {

  'a': ['1', '2', '3'],

  'b': ['2', '3', '4'],

  'c': ['3', '5', '6'],

}



var result = Object.keys(arr).reduce(function(r, e) {

  arr[e].forEach(function(a) {

    if (!r[a]) r[a] = [];

    r[a].push(e)

  })

  return r;

}, {});



console.log(result)
