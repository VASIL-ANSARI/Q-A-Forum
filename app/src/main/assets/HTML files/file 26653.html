You're missing a final }); (and a few commas) in your code example. It should look like this:

<script>
$(function(){
    $('div#tabs ul li a').click(function() {
        var n = $(this).attr('href');
        var p = n.slice(5, 6);
        alert(p);
        $.ajax({
            type: 'post',
            url: 'test.php',
            data: {value : p},
            success: function(data) {
                // Do something with response 'data'
            }
        });
    });
});
</script>
