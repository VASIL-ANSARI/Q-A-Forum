Use .attr() or prop() to set form action and closest() to get the form. Also notice the use of $(this) instead of $('#insert').

$('#insert').click(function () {
    var isAdd = $(this).text() == 'Add';
    $(this).text(isAdd ? 'Cancel' : 'Add');
    $(this).closest('form').attr('action', isAdd ? 'update.php' : 'add.php');
});
