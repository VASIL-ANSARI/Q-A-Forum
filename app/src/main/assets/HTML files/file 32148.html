double *copy_and_reverse(double *a, int length){
    double *result = new double[length*2];
    for(int i=0;i<length;i++) {
        result[i]=a[i];
    }
    int r = length*2;
    for(int i=0; i < length;i++){
        result[--r]=a[i];
    }
    return result;
}
