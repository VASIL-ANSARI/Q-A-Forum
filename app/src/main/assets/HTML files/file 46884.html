Your regex is wrong.

var str = '[jon Ned Brian Sam]';

str = str.replace(/\[|\]/g, '');

alert(str);

OR, 

var str = '[jon Ned Brian Sam]';

str = str.replace('[', '').replace(']', '');

alert(str);

OR,

you can use substr methods.

var str = '[jon Ned Brian Sam]';

str = str.substr(1, str.length - 2);

alert(str);
