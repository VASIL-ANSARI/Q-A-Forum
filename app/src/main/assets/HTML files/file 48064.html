Basically, just have a member of your comparison function type and have a constructor that takes a reference to that type so that the object can be passed a function pointer/lambda/std::function if needed. Here's how the GNU libstdc++ does it for std::priority_queue (with the irrelevant bits stripped away):

template <class _Tp, class _Sequence = vector<_Tp>, 
          class _Compare = less<typename _Sequence::value_type> >
class  priority_queue {
protected:
  _Sequence c;
  _Compare comp;
public:
  explicit priority_queue(const _Compare& __x) :  c(), comp(__x) {}
