def solve(a, k):
    tbl = {0 : -1}
    sum = 0
    n = len(a)

    for i in xrange(n):
        sum = (sum + a[i]) % k
        if sum in tbl:
               key = tbl[sum]
               result = a[key+1: i+1]
               return result
        tbl[sum] = i
    return []
