The issue here is caused by the fact that JS regex engine does not advance its index when matching an empty string. You can move it "manually" with some additional code (taken from regex101.com sample code generator page):

var re = /^$/g; 

var str = '';

if ((m = re.exec(str)) !== null) {

    if (m.index === re.lastIndex) {  // Here you manually advance

        re.lastIndex++;              // the index

    }

    alert(m[0]);

}
