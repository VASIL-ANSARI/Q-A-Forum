You can use itertools.combinations:

from itertools import combinations

data = [['A', 'B', 'C', 'D'],
        ['E', 'F', 'G'],
        ['I', 'J']]

result = []
for sublist in data:
    result.extend(map(list, combinations(sublist, 2)))

print result

OUTPUT 

[['A', 'B'], ['A', 'C'], ['A', 'D'], ['B', 'C'], ['B', 'D'], ['C', 'D'], ['E', 'F'], ['E', 'G'], ['F', 'G'], ['I', 'J']]
