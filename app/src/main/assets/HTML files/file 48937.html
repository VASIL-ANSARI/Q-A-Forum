Delegating constructor to the rescue:

class Owner {
public:
    Owner(std::unique_ptr<Member> m) : Owner(m, m.get()) {}

private:
    Owner(unique_ptr<Member>& m, Member* ptr) : member(std::move(m)){
        ptr->setOwnership(this);
    }

protected:
    const std::unique_ptr<const Member> member;
};
