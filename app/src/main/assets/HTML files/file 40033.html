Use for-else loop:

primes = set([1,2,3])

for r in range(1,20):

    if (r not in primes):
        for q in range(2,r):
            if r % q == 0:
                break
        else:
            #this will execute only if the for-loop completed without any breaks 
            primes.add(r)

print primes
