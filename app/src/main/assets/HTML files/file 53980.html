You can try css3 filter "invert". It is not really a very good result by now, but maybe you will play with it.

$(document).ready(function() {

  $("h2").each(function() {

    var offset = $(window).scrollTop() - $(this).position().top;

    var height = $(document).height();

    var percent = (-offset * 100) / height;

    $(this).css({

      'filter': 'invert(' + percent + '%)'

    })

  });

});
