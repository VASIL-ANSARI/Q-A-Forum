"Object has no method 'setDataSource'" occurs when the .swf is not fully loaded.

This function in javascript will set the Data Source until its fully loaded thruogh setInterval:

function setAttribute(){
          var attributes = {
           id: "grupos",
            name: "grupos"
          };
          swfobject.embedSWF("grupos.swf", "grupos", "1025", "678", "9.0.0", attributes);
           var i = 1;
              var sInt = setInterval(function() {
                if(i >= 10) {
                  clearInterval(sInt);
                }else{
                  i++;
               }
               document.getElementById("grupos").innerHTML = "<object type=\"application/x-shockwave-flash\" data=\"SWFObjects/test_" + i + ".swf\" width=\"300\" height=\"120\">";
