Use the change event:

$( "#slider" ).slider({
    value:0,
    min: 0,
    max: 2,
    step: 1,
    slide: function( event, ui ) {
        $( "#amount" ).val( "$" + ui.value );
    },
    change: function( event, ui) {
        if (ui.value == 0) {
            $("#mindiv").show();
        } else {
            $("#mindiv").hide();
        }
    }
});
