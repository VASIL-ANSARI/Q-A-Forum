You should modify your script, because $(document).ready() doesn't work with update panel:

function InitCss() {

        $(".Accordion").accordion({
            heightStyle: "content"
        });

        $(".special").button();

        $(".radio").buttonset();

        //$(".tabs").tabs();
        $(".tabs").tabs();

    }

Sys.WebForms.PageRequestManager.getInstance().add_pageLoaded(InitCss);

The same from code-behind:

protected void Page_Load(object sender, EventArgs e)
    {
 if (!this.Page.ClientScript.IsStartupScriptRegistered("InitCss"))
        {
            string script = @"function InitCss() {

        $("".Accordion"").accordion({
            heightStyle: ""content""
        });

        $("".special"").button();

        $("".radio"").buttonset();

        //$("".tabs"").tabs();
        $("".tabs"").tabs();

    }

Sys.WebForms.PageRequestManager.getInstance().add_pageLoaded(InitCss);";
            ScriptManager.RegisterStartupScript(Page, Page.GetType(), "InitCss", script, true);
        }
}
