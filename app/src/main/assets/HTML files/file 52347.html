You need to use setStartDate to set start date after datepicker was initialized

Here's jsFiddle

 $('.datepick').datepicker({

     autoclose: true,

     todayHighlight: true,

     /* startDate: '+1m'  This option will work here*/

 });

 

  $('#mydate').datepicker('setStartDate','+1m');
