There is no true pass-by-reference in JS. However, I came up with this trick (objects are the only things in JS that are passed by reference):

var white = { val: 1 },
    black = { val: 15 },
    flag = 'white', color;

if (flag == 'white') {
  color = white;
  color.val++;
} else {
  color = black;
  color.val--;
}

alert(white.val + ' ' + black.val); // 2, 15
