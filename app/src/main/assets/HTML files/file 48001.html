    for ($i = $key + 1 ; $i < sizeof($array); $i++) {
        if ($value['name'] === $array[$i]['name'] && $value['date'] === $array[$i]['date']) {
            $array[$key]['duplicate'] = 'TRUE';
            $array[$i]['duplicate'] = 'TRUE';
        }
    }
