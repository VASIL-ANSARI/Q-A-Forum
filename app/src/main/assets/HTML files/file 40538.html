A solution in bash:

#!/bin/bash
for file in `ls`
do
    ./program $file
done

Of course you'd have to modify your main function to take an argument to pass to the fstream constructor:

int main(int argc, char **argv)
{
    if (argc != 2)
    {
        // some error handling code
    }

    ifstream templateData(argv[1]);
    if (!templateData)
    {
        // more error handling
    }

    // process the file
}
