There should be no difference between a computer move and a player move. It is just a move with a color. For the user, you ask for input [of the color]; for the computer, you calculate a move [of the color], for example:

int numbermoves = 0;
char playerColor;

printf("What color do you want (W/B)?");
scanf ("%c", &playerColor);
do {
    if ((numbermoves%2)==0) { // white
        if (playerColor=='W')
             askMove ('W', &move);
        else calcMove('W', &move);
        putMove('W', &move);
    }
    else {
        if (playerColor=='B')
             askMove ('B', &move);
        else calcMove('B', &move);
        putMove('B', &move);
    }
    numbermoves++;

} while (numbermoves < size * size);
