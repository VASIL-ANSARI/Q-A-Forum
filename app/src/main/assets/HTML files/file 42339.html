I discovered my issue was related to photos being added after page load, rather than anything having to do with the image path. I was initializing 'Magnific Popup' on page load with the following script

$('.photo').magnificPopup({
      type: 'image',
      gallery:{
        enabled:true
    },
});

Here's the code I used to add the image to my page from the device:

function onSuccess(imageURI) {                                        
    var holder = document.getElementById('myImage');
    var image = '<div class="edit-img" ><img class="remove" src="images/remove.png" />&nbsp;<a class="photo" href="' + imageURI + '"><span class="img-name">' + imageURI.lastIndexOf("/") + 1 + '</span>&nbsp;<img class="img-thumb" src="' + imageURI + '" /></a></div>'
    $(holder).after(image);
}

The solution was to add the Magnific Popup code inside the function onSuccess(). My final code looks like this:

function onSuccess(imageURI) {                                        
    var holder = document.getElementById('myImage');
    var image = '<div class="edit-img" ><img class="remove" src="images/remove.png" />&nbsp;<a class="photo" href="' + imageURI + '"><span class="img-name">' + imageURI.lastIndexOf("/") + 1 + '</span>&nbsp;<img class="img-thumb" src="' + imageURI + '" /></a></div>'
    $(holder).after(image);

    $('.photo').magnificPopup({
        type: 'image',
        gallery:{
          enabled:true
        },
    });
}
