<div class="form-group">  
  <label class="col-md-4 control-label" for="{{ field.label }}"> {{ field.label }}</label>
  <div class="col-md-6">
    {{ field }}
    {% for error in field.errors %}

      <strong>{{ error|escape }}</strong>
    </p>
  {% endfor %}
  </div>
