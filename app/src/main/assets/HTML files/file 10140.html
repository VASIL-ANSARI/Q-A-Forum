    $imgname = "source.png";           
    $im = imagecreatefrompng ($imgname);              
    if($im && imagefilter($im, IMG_FILTER_COLORIZE, 137, 113, 79))               
    {
         echo 'Image successfully shaded green.';                

         imagepng($im, 'result.png');             
         imagedestroy($im);                     
    }                  
   else               
   {
