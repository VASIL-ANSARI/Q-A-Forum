You could do something like this:

$times = array(
    144441415=>"8:00am", 
    1444784744=>"8:30am",
    1444784745=>"8:45am",
    1444784746=>"9:00am",
    1444784747=>"10:00am",

);

foreach ($times as $time) {
    if (date("H:00:00", strtotime($time)) == date("H:i:00", strtotime($time))) {
        echo '<br>Whole Hour: '.$time;
    }
    else {
        echo '<br>Not Whole Hour: '. $time;
    }
}
