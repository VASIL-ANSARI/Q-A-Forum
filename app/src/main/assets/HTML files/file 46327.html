    <script type="text/javascript">

    $(document).ready(calculate());   
    $(document).on("keyup", calculate());

    function calculate() {
        var sum = 0;
        $(".totalCal").each(function(){
            sum += +$(this).val();
        });
        $("#total").val(sum);

        document.getElementById('grandTotal').innerHTML = sum.toFixed(2);
        });
    }
</script>
