You are doing it wrong way since Case Has many Evidences .. One option is you have to implode array with ,

try,,

<?php
$evidences  = \yii\helpers\ArrayHelper::getColumn((\yii\helpers\ArrayHelper::getColumn($model , 'evidences')) ,'evidence_type');
?>
<?= \yii\widgets\DetailView::widget([
    'model' => $model,
    'attributes' => [
        'ref_no',
        'case_description',
        'raised_on',
        'status',
        'updated_on',
        [
            'label' => 'Evidences',
            'value' => implode(',' ,$evidences),
        ],
        'evidences.evidence_type',
    ],
]) ?>
