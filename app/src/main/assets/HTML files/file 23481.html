using namespace cv;
using namespace std;
int main( int argc, char** argv )
 {
IplImage *src,*dst;
src=cvLoadImage("img.png");
cvNamedWindow("SRC", CV_WINDOW_AUTOSIZE);
cvNamedWindow("DST", CV_WINDOW_AUTOSIZE);


 dst = cvCreateImage(cvSize(500,500),src->depth,src->nChannels);
cvResize(src,dst);


cvShowImage("SRC",src);
cvShowImage("DST",dst);


cvWaitKey(0);
cvDestroyWindow("SRC");
cvDestroyWindow("DST");
cvReleaseImage(&dst);
cvReleaseImage(&src);

return 0;
}
