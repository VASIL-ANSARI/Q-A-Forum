CREATE TABLE words(id INT, word VARCHAR(100));

INSERT INTO words
SELECT 1 , 'arse' UNION ALL SELECT 2,'ball' UNION ALL SELECT 3, 'sex';

Query:

SELECT w.word, COUNT(*) AS `occ`
FROM posts p
JOIN words w
 ON p.post LIKE CONCAT('%', w.word, '%')
WHERE p.id = 1
GROUP BY w.word
ORDER BY occ DESC;
