If the object has the attribute length defined it should be an array. Be careful about the comparison because if (result.length) will be false when the array is empty. However this function will be much more useful, found in this answer JavaScript: Check if object is array?

window.isArray = function(candidate){
    if (candidate == null)
        return false;
    return ( Object.prototype.toString.call(candidate) === '[object Array]' )
}
