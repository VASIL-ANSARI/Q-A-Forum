<script>
$("#prod_cat").change(function(){
    //get category value
    var cat_val = $("#prod_cat").val();
    // put your ajax url here to fetch subcategory
    var url             =   'www.example/fetch_subcategory.php';
    // call subcategory ajax here 
    $.ajax({
                   type:"POST",
                   url:url,
                   data:{
                       cat_val : cat_val
                   },

                   success:function(data)
                    {
                        $("#sub_cat").html(data);
                    }
                });
});
</script>
