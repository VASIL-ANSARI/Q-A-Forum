Hopfully, i am able to achieve the result by using combobox as the type in my custom button.

setup: function (editor) {
        editor.addButton('mybutton', function() {
           var items= ["8", "9", "10", "11", "12", "14", "16", "18", "20", "22", "24", "26", "28", "36", "48", "72"]
           return {
                  type: 'combobox',
                  autofocus: true,
                  label: 'FontSizes',
                  values: items,
                  onChange: function(e){
                        console.log(this.value());
                  }
           }
       }
