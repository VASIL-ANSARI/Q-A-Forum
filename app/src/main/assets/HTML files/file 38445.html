  function AlbumCtrl($scope,$timeout) {
      $scope.counterArray = [5,10,15];
      $scope.counter = 5;

      $scope.index = 0;

      $scope.onTimeout = function() {
          $scope.counter = --$scope.counterArray[$scope.index];

          if ($scope.counterArray[$scope.index] === 0 && $scope.index < 3) {
              $scope.index++;         
          }

        $timeout($scope.onTimeout, 1000);

      };



    $timeout($scope.onTimeout,1000);

  }
