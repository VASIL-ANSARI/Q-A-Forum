$(function () {
    var $i = $("#gg")
    $i.click(function () {
        $i.css('pointer-events','none');
        $(document).on('mousemove', function (e) {
            $i.css({
                left: e.pageX - 42,
                top: e.pageY - 60
            });
        });
        $('#gg1').one('mouseenter', function() {
            $(this).hide();
        });
    });
});
