If I keep your example, you can access the price property of the superclass during initialization of the childclass and set a default value for money like this:

class EFPaymentObject: NSObject {

    var price: NSString = "1200"
    var title: NSString?
    var desck: NSString?
    var hasPass: Bool?

}

class EFGoodsPayment: EFPaymentObject {

    var pay: NSString = "1"
    var type: NSString = "3"
    var goods: NSString = ""
    var money: NSString?

    override init() {
        super.init()
        money = super.price
    }

}


let payment = EFGoodsPayment()
println(payment.money!)       // prints "1200"
