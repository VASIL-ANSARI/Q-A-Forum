You could find the corresponding tr after you post and remove it:

$( ".del" ).click(function() {
  var id = $(this).attr( "id" )
  $.post( "admin_delete.php", { reqidno: id })
   .done(function( data ) {
    alert( data );
    // find the clicked button
    // take the closest 'tr' and remove it
    $("#"+id).closest('tr').remove();
 });
});
