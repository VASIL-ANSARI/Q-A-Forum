Looks like jQuery, so removing the inline event handler and adding a proper one is better.
Then check the audio.paused property so if the audio is paused

$('#btn_pause').on('click', function(e) {
    e.preventDefault();

    var audio = $('#audioLayerControl').get(0);

    if (audio.paused) {
        audio.play();
    } else {
        audio.pause();
    }

});
