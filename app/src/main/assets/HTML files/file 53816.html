One way of doing it:

import datetime
def myFun(year):
     res = []
     d = datetime.datetime(year, 1, 1)
     while d.year != year +1:
             res.append(d)
             d = d + datetime.timedelta(days=1)
     return res

>>> myFun(2014)[:2]
[datetime.datetime(2014, 1, 1, 0, 0), datetime.datetime(2014, 1, 2, 0, 0)]
>>> len(myFun(2014))
365
