One way to do it is to setup Roles and then use code like this in your views:

<%  
    if (User.IsInRole("AdminRole") 
        Html.RenderPartial("AdminView");  
    else if (User.IsInRole("SalariedRole") 
        Html.RenderPartial("SalaryView");  
%>
