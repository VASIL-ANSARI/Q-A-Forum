One option is to use a join for that:

update `table` t
   join (select '160009' somevalue, 'val1' othervalue
         union all
         select '160010' somevalue, 'val2' othervalue
         union all
         select '160011' somevalue, 'val3' othervalue
         ) t2 on t.column1 = t2.somevalue
set t.column2 = t2.othervalue
