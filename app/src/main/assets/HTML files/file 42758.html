from turtle import*

def file():
    filename = "t.txt"
    openfile = open(filename, "r")
    readlines = openfile.readlines()
    nums=[]
    for lines in readlines:
        nums.append(lines.strip().split(","))

    return nums

def turtle(n):
    print(n)
    row = 0
    col = 0
    wn = Screen()
    pen = Turtle()
    wn.setworldcoordinates(-0.5,3.5,4.5,-1.0)

    pen.up()
    for numberRow in n:
        for number in numberRow:
            pen.goto(col,row)
            pen.write(number, font=("arial", 30))
            col+=1
        row += 1
        col = 0

def main():
    y = file()
    w = turtle(y)


main()
