You can put it on the body, onload event..

or

If you want to keep things all javascript, you can also use document.addEventListener('DOMContentLoaded', funciton())

example using body.onload

function move() {

  var elem = document.getElementById("myBar");   

  var width = 10;

  var id = setInterval(frame, 1200);

  function frame() {

    if (width >= 100) {

      clearInterval(id);

    } else {

      width++; 

      elem.style.width = width + '%'; 

      document.getElementById("label").innerHTML = width * 1  + '%';

    }

  }

}
<body onload="move()">

  <div id="myProgress">

    <div id="myBar">

      <div id="label">10%</div>

    </div>

  </div>

</body>
