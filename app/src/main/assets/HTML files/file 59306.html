$(document).on('click', '.edit-city-link', function(){
  var client_id = $(this).data('client-id');
  var client_name = $(this).data('client-name');
  $('#client-id').val(client_id);
  $('#client-name').val(client_name);
  $('#edit-client-modal').modal('show');
});
