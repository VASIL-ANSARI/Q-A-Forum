Since that's not syntactically valid JavaScript (JSLint could have told you that), no, it will not work.

$.extend is your friend here. Try this (not tested):

// Define rules common to all forms
var commonRules = { 
  email: {
    required: true,
    email: true
  },
};

// Define rules for form A
$("#formA").validate({
   rules:  $.extend({
     uniqueFieldA: "required"
   }, commonRules)
});

// Define rules for form B
$("#formB").validate({
   rules: $.extend({
     uniqueFieldB: "required"
   }, commonRules)
});

You may need to perform a recursive merge (a "deep extend") if there is any overlap between the common rules and the form-specific ones.
