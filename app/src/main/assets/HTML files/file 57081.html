You can simply iterate through the array and build your object property-by-property.
It can be done easier with Array.prototype.forEach:

var arr = [

  {

    "name": "foo",

    "value": "bar"

  },

  {

    "name": "foo1",

    "value": "bar1"

  }

];



var o = {};



arr.forEach(function(x) {

  o[x.name] = x.value;

});



console.log(o);
