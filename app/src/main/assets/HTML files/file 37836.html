$("#date2").datepicker({
    dateFormat: "yy/mm/dd",
    changeMonth: true,
    changeYear: true,
    minDate: $('#date').val(),
    onSelect:function(date){
        var start_date = new Date($('#date').val());
        var end_date = new Date(date);
        if(start_date > end_date ){
            alert('date start should be less than date end');
            $(this).val('');
            //$('#date').val(''); // add if you want
        }
    }
});
