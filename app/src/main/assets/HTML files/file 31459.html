$('.picture').on('mouseenter', function() {
    $('.picture').css({top: '0', left: '200px'});
    $(this).css({top: '60px', left: '20px'});
});

$('.picture').on('mouseleave', function() {
    $('.picture').css({top: '0', left: '200px'});
    $(this).css({top: '0', left: '200px'});
});
