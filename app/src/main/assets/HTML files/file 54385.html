You can Create your own SearchView class by extending SearchView like so:

public class DemoSearchView extends SearchView {


    public DemoSearchView(Context context){
        super(context);

        //.. Do your setup
    }


    @Override
    public void onActionViewExpanded() {
        super.onActionViewExpanded();

        //.. Do the work you need
    }

    @Override
    public void onActionViewCollapsed() {
        super.onActionViewCollapsed();

        //.. In case you want to listen for collapse
    }
}
