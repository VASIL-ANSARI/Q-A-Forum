                //Called Every Hit
                public void ShockerGeneratorPlayerOne()
                {
                    Random rnd = new Random();
                    RanShocMatch = rnd.Next(10);
                    if (RanShocMatch == 1)
                    {
                    //Speed Boost!
                        VelocityX = (VelocityX - 1);
                        VelocityY = (VelocityY - 1);
                    }
                    else if (RanShocMatch == 2)
                    {
