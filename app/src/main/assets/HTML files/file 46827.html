    background: orange;

}

    margin: 0;

    padding: 0;

    list-style-type: none;

}



nav > ul > li {

    position: relative;

    top: 0;

    left: 0;

    margin-left: 50px;

}



nav > ul > li > a {

    display: inline-block;

    color: white;

    text-decoration: none;

    padding: 5px 10px;

    background: green;

}



nav > ul ul {

    background: #eee;

    border: 1px solid #000;

    padding: 10px;

    position: absolute;

    left: -10px;

}
