The modern way of doing this is to use something like BeautifulSoup. One does not try to process mark up using regular expressions anymore because the resulting code is unreadble and even the slightest change to the markup can break your beautifully crafted regular expression.

from bs4 import BeautifulSoup

bs = BeautifulSoup('<tr class="data"><td class="first"><a href="en/clinics/al-khayal-medical-centre-6d15db1f.aspx">AL KHAYAL MEDICAL CENTRE</a></td><td class="second not-responsive"><a href="/portal/en/healthcare/clinics.aspx?speciality=Urology">Urology</a></td><td class="third"><label style="display:block ;direction:ltr; text-align:left">042765600</label></td><td class="fourth"><a href="http://www.alkhayalmedicalcentre.com" target="_blank">www.alkhayalmedicalcentre.com</a></td><td class="fifth not-responsive"><a class="clinic-details btn noborder basic small" href="en/clinics/al-khayal-medical-centre-6d15db1f.aspx" rel="1dfd7cf2-49c8-43eb-80b8-9393c9f48ca4">More Info</a></td></tr>   ')

for a in bs.find_all('a'):
   print a['href']
