void find_minimum(double a[], size_t n) {

  size_t i, index;
  double smallest;

  // sanity check to make sure we're not accessing outside allocated memory
  if (n > 0) {
    smallest = a[0];
    index = 0;
  } else {
    // nothing left to do here
    return;
  }

  // start at 1 because 0 is initial default
  for (i = 1; i < n; i++) {
    if (a[i] < smallest) {
      smallest = a[i];
      index = i;
    }
  }

  // assign only the smallest index
  a[index] = 0.0;
}
