I defined the error500.jsp as a servlet with an url pattern and it's now working with the config in web.xml.

But a direct access to error500.jsp get me a 404. So there is a problem but there is a work around.

<servlet>
    <servlet-name>error500</servlet-name>
    <jsp-file>/WEB-INF/jsp/error500.jsp</jsp-file>
</servlet>

<servlet-mapping>
    <servlet-name>error500</servlet-name>
    <url-pattern>/error500</url-pattern>
</servlet-mapping>

<error-page>
    <error-code>500</error-code>
    <location>/error500</location>
</error-page>
