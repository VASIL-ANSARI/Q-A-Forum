Try this,

$("#s").hide(0);
$("#u").hide(0);

$(document).click(function (e) {
    if (!$(e.target).hasClass('popu')) {
        $("#s,#u").hide(200);
    }
});
$("#e").click(function (e) {
    $("#s").show(200);
    e.stopPropagation();
});

$("#er").click(function (er) {
    $("#u").show(200);
    er.stopPropagation();
});
