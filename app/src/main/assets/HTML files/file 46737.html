your_controller.rb

def update_nearby
  find_nearby_shows
end

private
def find_nearby_shows
  @radii = params[:radii] ? params[:radii] : 50
  @nearby_venues = Venue.near("Boulder, CO",@radii,:select =>
    "id").to_a
  @nearby_shows = Show.where(show_venue: @nearby_venues)
end
