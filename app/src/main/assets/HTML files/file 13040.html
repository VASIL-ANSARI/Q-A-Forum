class AddValRequest {
    public $fVal1;
    public $fVal2;
}

class AddValResponse {
    public $fResult;
}

function AddVal(AddValRequest $request) {
    $response = new AddValResponse();
    $response->fResult = $request->fVal1 + $request->fVal2;
    return $response;
}

$server = new SoapServer("numops.wsdl");
$server->addFunction("AddVal");
$server->handle();
