namespace bio = boost::iostreams;

void decompress(unsigned char const* arr, int nCopySize, std::vector <unsigned char>& vecArg)
{
    using namespace boost::iostreams;

    bio::stream<bio::array_source> arrs(reinterpret_cast<char const*>(arr), nCopySize);

    bio::filtering_streambuf<bio::input> iin;
    iin.push(bio::gzip_decompressor());
    iin.push(arrs);

    std::vector<char> tmp;
    boost::iostreams::copy(iin, bio::back_inserter(tmp));

    vecArg.assign(tmp.begin(), tmp.end());
}
