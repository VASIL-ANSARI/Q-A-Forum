One way of achieving this would be to take advantage of the vector range constructor and construct each vector from a pair of iterators to the start and end of each "row" in the list.

std::vector<std::vector<double>> v2;
auto begin_row = matrix2.cbegin();
auto end_row = std::next(begin_row, n);
for (int i = 0; i != n; ++i, begin_row = end_row, std::advance(end_row, n))
  v2.emplace_back(begin_row, end_row);

Edit: Or here is another way, create a vector of the correct size and then loop over every index, this one works in C++03:

std::vector<std::vector<double> > v2(n, std::vector<double>(n));
std::list<double>::const_iterator itr = matrix2.begin();
for (size_t i = 0; i != n; ++i)
  for (size_t j = 0; j != n; ++j)
    v2[i][j] = *(itr++);
