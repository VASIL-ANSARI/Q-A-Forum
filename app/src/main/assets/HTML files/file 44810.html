This is cheesy and untested, but assuming you have your DIVs in a parent container, I'd find them all, and their position after animation, sort on position and then take the highest one off the list. Like this:

var divs = $('parent > div').map(function(i, o) { return [o.position().left, o]; });
divs.sort(function(a, b) { return a[0] < a[1] ? -1 : a[0] > a[1] ? 1 : 0; });
var highest = divs.pop()[1];
