$("#clearEnd").click(function() {
    if($display.val().length > 1){
        $display.val($display.val().substring(0,$display.val().length-1));
        // ex) 123->12, 1000->100
    }else{
        // this means click clear[C] button
        $("#clear").trigger("click");
        // ex) 1->0, 2->, 0->0
    }
});
