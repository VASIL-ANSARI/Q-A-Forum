<script>
        var client = new XMLHttpRequest();
        client.open('GET', 'load_lines.txt');
        client.onreadystatechange = function() {
                t = client.response 
                t = t.split('\n');
                var listbox;
                for(i=0;i<t.length;i++){
                        listbox +='<option value="'+i+'">'+t[i]+'</option>';
                }
                document.getElementById('lb').innerHTML=listbox;
        }
        client.send();
</script>
