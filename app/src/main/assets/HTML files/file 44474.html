Depending on your needs, you can use simple array, or object, with key:value pair:

arr=[];
$( ".result" ).each(function( index ) {
 key=$( this ).prop('id');
 value=$(this).val();
    arr.push(key+':'+value);   
});

console.log(arr);

obj={};

$( ".result" ).each(function( index ) {
 key=$( this ).prop('id');
 value=$(this).val();

    obj[key]=value; 
});

console.log(obj);
