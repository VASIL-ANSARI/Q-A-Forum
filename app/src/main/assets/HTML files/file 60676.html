std::vector<int> result;
for (int i = 0; i<MAXL;i++)
{
    for (int j=0; j<MAXC;j++)
    {
        if (image1[i][j] != BackColor)  // Notice !=
        {
            result.push_back(i+1);
            result.push_back(j+1);
            result.push_back(image1[i][j]);
        }
    }
}


for (auto x : result)
{
    std::cout << x << " ";
}
std::cout << endl;
