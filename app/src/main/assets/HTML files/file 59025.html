First of all in Swift use generic type Dictionary like [String:Any] instead of NSDictionary and after you get dictionary get the array from the value key and iterate through the Array and get the wantThisOne from every object.

if let jsonResult = try JSONSerialization.jsonObject(with: data!, options: []) as? [String:Any] {
     if let valueArray = jsonResult["values"] as? [[String:Any]] {
          for item in valueArray {
              print("Date - \(item["wantThisOne"])")
          }
     }
}
