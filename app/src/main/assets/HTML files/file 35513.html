function get_unique_array($array)
{
   $result = array_map("unserialize", array_unique(array_map("serialize", $array)));

   foreach ($result as $key => $value)
   {
      if ( is_array($value) )
      {
        $result[$key] = get_unique_array($value);
      }
   }

   return $result;
}
