Assuming graph: and something are separated by whitespace

void load2()
{
  ifstream infile("graph.txt");
  ofstream outfile("out.txt");

  std::string graph, something;

  while (infile >> graph >> something) {
    if (graph != "graph:") {
      break;
    }
    outfile << something;
  }
}
