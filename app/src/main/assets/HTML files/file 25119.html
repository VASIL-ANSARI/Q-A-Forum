var lastScrollTop = 0;
$(window).scroll(function(event){
   var st = $(this).scrollTop();
   if (st > lastScrollTop){
       // downscroll code
       window.location.hash='musection2';
   } else {
      // upscroll code
      window.location.hash='musection1';
   }
   lastScrollTop = st;
});
