I had no solution on the forum using Carbonfive ssTempalte's API, I implemented a workaround. I get the marshelled HSSFWorkbook object and parse the list of its sheets. If it contains the string _HIDDEN I put it in the list of hidden sheets of workbook.

public HSSFWorkbook build() {
        HSSFWorkbook workbook = null;
        try {

            Map<String, Object> context = new HashMap<String, Object>();
            context.put("msg", messageResources.getMessages());
            populateContext(context);

            SsTemplateProcessor processor = SsTemplateProcessor.getInstance();
            workbook = processor.process(new ClassPathResource(getTemplatePath()).getFile(), context);

            // For hide sheets
            if (needHideSheet) {
                int index = 0;
                while (index < workbook.getNumberOfSheets()) {
                    String sheetName = workbook.getSheetName(index);
                    if (StringUtils.contains(sheetName, "_HIDDEN")) {
                        workbook.setSheetHidden(index, true);
                    }
                    index++;
                }

            }

        } catch (Exception e) {
            logger.error("Error during marshalling of workbook template", e);
        }
        return workbook;
    }
