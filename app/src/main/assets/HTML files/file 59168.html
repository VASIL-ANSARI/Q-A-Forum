You only have 3 new bars, so the number of elements on your data has changed. You need to use the update pattern.

var rects = plot.selectAll("rect")
  .data(currentDatasetBarChart);

  rects.enter()
   .append("rect")
    //Code to style and define new rectangles.

  //Update 
  rects.update()
  .transition()
  .duration(750)
  .attr("x", function (d, i) {
    return xScale(i);
  })
  .attr("width", width / currentDatasetBarChart.length - barPadding)
  .attr("y", function (d) {
    return yScale(+d.measure);
  })
  .attr("height", function (d) {
    return height - yScale(+d.measure);
  })
  .attr("fill", colorChosen);

  // Remove unused rects
  rects.exit().remove();
