I'm assuming you want NaN's every few minutes, not just one NaN in the gap, and that you don't mind adding NaN's where there is no gap, as long as they're also added in the gap. Let me know if this solution does what you want:

# Imports

from datetime import datetime

import numpy as np
import pandas as pd

# Setup

initial_index = [datetime(2014,4,21,12,x) for x in [0,5,8,14,18,21,25]]
columns = ['A','B','C','D']

df = pd.DataFrame(np.random.randn(7,4), index=initial_index, columns=columns)

# The actual solution

regular_interval_index = pd.date_range('12:00:00', '13:00:00', freq='5Min')

df_reindexed  = df.reindex(df.index + regular_interval_index)
