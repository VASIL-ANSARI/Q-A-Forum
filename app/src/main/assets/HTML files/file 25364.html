Just simply use ::sleep() it isn't in Qt it's a POSIX.1-2001 function.

Also you can try this code, because QThread::sleep() calls ::Sleep()

class mythreadhelper : public QThread
{
   public:
   static void mysleep(int ms)
   {
      return sleep(ms);
   }
};
