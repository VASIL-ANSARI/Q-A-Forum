As mentioned in the comments, you can implement an indexer for your class Locale.

Example

public class Locale
{
    Dictionary<string, string> _dict;
    public Locale()
    {
        _dict = new Dictionary<string, string>();
        _dict.Add("dot", "net");
        _dict.Add("java", "script");
    }
    public string this[string key] //this is the indexer
    {
        get
        {
            return _dict[key];
        }
        set //remove setter if you do not need
        {
            _dict[key] = value;
        }
    }
}

Usage:

var l = new Locale();
var value = l["java"]; //"script"
