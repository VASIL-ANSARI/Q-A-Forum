try this

     SELECT `user_id`,
     max(case when  `meta_key` = 'name' then `usermeta` end)  as name  ,
     max(case when  `meta_key` = 'userurl' then `usermeta` end) as website ,
     max(case when  `meta_key` = 'points' then `usermeta` end ) as points 
     FROM wp_usermeta
     GROUP BY `user_id`           
     ORDER BY points DESC
