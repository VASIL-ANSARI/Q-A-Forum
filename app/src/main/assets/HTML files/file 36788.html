The below would never have worked

i[key].Date.getMonth();

As your Date property would have been a string and not a Date object so it would have stopped due to an undefined function error.

The below will accumulate the data, but instead of having an object with a single property named "0" containing the data, this will create an object with properties of {month} {year}, i.e. Aug 2014, Jul 2014, where the data is contained

var data = {};

//Loop over the array, use a regular for loop or
//forEach polyfill if supporting older browsers
i.forEach(function(item,index){
   //Since the object we want is a property
   //of the item object use the index as the key
   var obj = item[index];
   var d = new Date(obj.Date);

   //alternative way to get abbr. month then creating
   //an array of months and using the getMonth as an index to that array
   var m = d.toDateString().split(" ")[1];
   var y = d.getFullYear();

   //Check to see if there already is a month/year 
   //property creating one if not.
   var temp = data[m+" "+y] || {"TV":0,"Video":0,"DVD":0,"Date":""};

   temp.TV += +obj.TV;
   temp.Video += +obj.Video;
   temp.DVD += +obj.DVD;
   temp.Date =  m+" "+y;

   data[m+" "+y] = temp;   
});
console.log(data);

Will log the data object, and will have the the following format:

{
    "Jul 2014": {
        "TV": 350,
        "Video": 110,
        "DVD": 300,
        "Date": "Jul 2014"
    },
    "Aug 2014": {
        "TV": 550,
        "Video": 70,
        "DVD": 250,
        "Date": "Aug 2014"
    }
}
