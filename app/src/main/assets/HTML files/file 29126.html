You can do something like:

$photos = Photo::orderBy('category')->orderBy('created_at', 'desc')->get();
return View::make('myView')->with('photos', $photos);

Then

<?php $category = ''; ?>
@foreach($photos as $photo)

    @if($category != $photo->Category)
        {{$photo->Category}}

    @endif

    {{ $photo->Title }}

@endforeach
