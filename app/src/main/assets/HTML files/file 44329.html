also, you need to return true from validateEmailAddress() when the email is valid:

function validateEmailAddress(address) {
    filter = /^\w+([\.-]?\w+)*@\w+([\.-]?\w+)*(\.\w{2,4})+$/;
    if (!filter.test(address)) {
        alert(address + ' - Is an invalid email address.');
        return false;
    }
    return true;
}
