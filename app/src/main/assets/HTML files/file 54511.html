#include <iostream>  
#include <fstream> 
#include <string>
using namespace std; 

int main()
{ 

  string Display = "";
  ofstream FileOut;
  ifstream FileInput;

  FileOut.open("C:\\Example.txt");

  FileOut << "This is some example text that will be written to the file!";

  FileOut.close();

  FileInput.open("C:\\Example.txt");

  if (!FileInput)
  {
      cout << "Error File not Found: " << endl; 
      return 1; 
  }

  while (!FileInput.eof())
  {
      getline(FileInput, Display);
  } 
  FileInput.close();
  cout << Display << endl; 

  return 0;
}
