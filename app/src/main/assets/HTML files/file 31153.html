 $('.well').on('keyup blur paste', '.balance', function() {
               var self = $(this);
           setTimeout(function() {
               var str = $(self).val();
               $("input[data-cost][debt]").data('cost',str.replace(/^\$/, ''));
               $("input[data-cost][debt]").data('debt',str.replace(/^\$/, ''));
               calculateTotalFor();
           }, 0)
});
