Create or Replace Trigger Late_Rule
BEFORE INSERT ON Transaction
For each Row
DECLARE

Fee Number;

BEGIN


If :new.Date_Returned > :new.Date_Due
THEN 
Fee := (:new.Date_Returned - :new.Date_Due) * 2;


END IF;

Update Member
    Set Unpaid_Bal = Unpaid_Bal + Fee;
end;
