Quick and dirty:

foreach ($arr as $key => $vals) {
    echo $key . implode(",", $vals) . "\n";
}

UPDATE:

$transformed = array();
foreach ($arr as $item) {
    foreach ($item as $key => $val) {
        $transformed[$key][] = $val;
    }
}
foreach ($transformed as $key => $vals) {
    echo $key . implode(",", $vals) . "\n";
}
