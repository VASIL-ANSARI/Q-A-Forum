You need to move your window width logic inside the scroll event. Currently you are attaching multiple scroll events when the screen is resized.

$(window).on("load resize", function () {

  $(this).scroll(function () {      

    if ($(window).width() > 752) {

      console.log("Scrolled in desktop version");

    }

    else {

      console.log("Scrolled in mobile version");

    }

  });    

});
