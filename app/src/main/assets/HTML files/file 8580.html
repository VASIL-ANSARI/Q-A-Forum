This is a N-ary tree. I suggest you of split in Tree and Node

typedef struct tree tree;
typdef struct node node;

struct tree {
    node * root;
};

struct node {
    int value;
    node * next_node;
};

now you can perform all the operation of the tree structure

here an example

node * add_child(node *parent, int child_value){
    node * chld = malloc(sizeof(node));
    chld->value = child_value;
    if(parent->next == NULL)
        parent->next = chld;
    else{
        node * tmp = parent->next;
        while(tmp->next != NULL)
            tmp = tmp->next;
        tmp->next = chld;
    }
    return chld;
}
