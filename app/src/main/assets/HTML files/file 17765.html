You need to check the change event of the select box, so when the value changes, you check again if something is selected and then change your button!

Also note the == in the if statement

$(document).ready(function() {
    $('#names').change(function() {
        if ($('#names').val() == '')
        {
            $('#jsbutton').attr('disabled', 'disabled');  
        }
        else
        {
            $('#jsbutton').removeAttr('disabled');
        }
    }
});
