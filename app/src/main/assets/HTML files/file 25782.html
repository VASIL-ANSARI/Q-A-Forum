I think the problem is the script is not able to reference sI from where you are calling clearInterval(sI); so you can make sI as a global variable . so declare sI as var sI=null at the top.

May be this is what you wanted.

var sI = setInterval(keepalive(), 30000); //place this here 

function validate() 
{
    var file = $("#file").val();
    if(!file || file == '' || file == null)
    {
        document.getElementById("outdata").innerHTML="Please select a file";
        return false;
    }   
    var options = {
          target: '#outdata',
          url:'process.php', 
          data:{
             accesstype:"cre"
          },
          beforeSubmit: function() {
             $('#uploader').html('<img src="/images/ajax-loader.gif" />');
             $('input[type=submit]').attr('disabled', true);
          },
          success:  function() {
              $('#uploader').html('');
              clearInterval(sI);//on success kill the set interval 
              $('input[type=submit]').attr('disabled', false);
          }
     };
     $('#cre').ajaxSubmit(options);
     return false;   
}
