I suppose you need to wire it all up...here is a sample I've quipped to explain the technique.

$(function() {

  var tabs = $('div .tab');

  var lis = $('li');

  $(tabs.selector)

    .filter(':odd')

    .css('background-color', '#abcdef');

  var ATTR = 'data-tab';

  var showTab = function(v) {

    tabs.hide();

    tabs.filter(function() {

      return $(this).attr(ATTR) == v;

    }).show();

  };



  $('a').each(function() {

    var el = $(this);

    if (el.parent().hasClass('active')) {

      showTab(el.attr(ATTR));

      return false;

    }

  }).click(function() {

    var el = $(this);

    tabs.hide();

    showTab(el.attr(ATTR));

    console.log(lis);

    //we have to set the li element

    //class as active...

    var a = lis.removeClass('active');

    var b = lis.filter(function() {

      return this === el.parent().get(0);

    }).addClass('active');

    console.log(b);

  });

});
