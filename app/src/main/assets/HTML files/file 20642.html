You can use SUBSTR, INSTR, and COALESCE, but there might be a better solution:

SELECT  COALESCE( SUBSTR(str, 0, INSTR(str, ' ')) , str) newStr
FROM (
  SELECT LTrim(Substr(STRING_NAME,INSTR(STRING_NAME,'me')+LENGTH('me'))) str
  FROM STRING_HELP 
  ) t;
