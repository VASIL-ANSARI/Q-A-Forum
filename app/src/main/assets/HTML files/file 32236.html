Looks like you've made a hash of the listener: you want to check the value of From, and update the values in To, but you're doing everything against To. Try this first for the listener:

FromComboBox.SelectedValueChanged += FromComboBox_SelectedValueChanged;

And check the values in FromComboBox:

void FromComboBox_SelectedValueChanged(object sender, EventArgs e)
{
    ToComboBox.Items.Clear();
    if (FromComboBox.SelectedItem.Equals("1"))
    {
        ToComboBox.Items.Add("O");
        ToComboBox.Items.Add("P");
    }
    else if (FromComboBox.SelectedItem.Equals("2"))
    {
        ToComboBox.Items.Add("H");
    }
}
