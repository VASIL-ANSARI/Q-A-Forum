I see that you place the click on an input, inputs make post back, probably you need to prevent that using the e.preventDefault(); as:

    $(document).ready(function () {
        $("input[id$=btnclick]").click(function (e) 
        {
            if ($("#content").html() != "") {
                // here prevent from post back
                e.preventDefault();

                AlertDialog($("#content").html());
            }
        });
    });
