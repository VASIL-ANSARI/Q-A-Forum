You can massively reduce this code by placing the selector of the divBasicDetails and divPersonelDetails into a variable which is set by the provided divHeaderId. You can also then join the selected div with divOtherDetails as you're performing the same logic on both elements, and you can use toggleClass in place of multiple addClass/removeClass calls. Try this:

function ToggleClose(divHeaderId) {
    MainDivClickCount = 1; // this can possibly be removed, as you don't appear to use it

    var idSelector = divHeaderId == '#divBasicDetails' ? '#divPersonelDetails' : '#divBasicDetails';
    var $divs = $(idSelector).add('#divOtherDetails');

    $divs.toggleClass('tabMinus tabPlus');
    $divs.find('img').attr('src', '../../Images/plus-icon.png');
    $divs.find('table tr td').toggleClass('tabOpen tabClose');
}
