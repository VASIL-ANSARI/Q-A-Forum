public class TeacherDetails 
{

You need to implement it like this;

private void button3_Click(object sender, EventArgs e)
{
    ProjectDBDataContext Context = new ProjectDBDataContext();
    var r =
            from b in Context.TeacherDetails
            where b.TeacherId <= 2
            select b; //Remove .First_Name here

    foreach (TeacherDetail b in r)            
        b.First_Name = "Jerry";

    Context.SubmitChanges();
}

You were selecting a string with select b.First_Name, whereas you would then loop over it as if it were a TeacherDetail in your foreach.
