To count how many times one word appears in a text:

text = "word aaa word word word bbb ccc ccc"
text.scan(/\w+/).count("word") # => 4

To count a set of words:

text = "word aaa word word word bbb ccc ccc"
wlist = text.scan(/\w+/)
wset = ["word", "ccc"]
result = {}
wset.each {|word| result[word] = wlist.count(word) }
result # => {"word" => 4, "ccc" => 2}
result["ccc"] # => 2
