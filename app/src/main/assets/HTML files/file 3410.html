>>> from itertools import combinations
>>>
>>> data = dict(
...   list1 = set(list("alphabet")),
...   list2 = set(list("fiddlesticks")),
...   list3 = set(list("geography")),
...   list4 = set(list("bovinespongiformencephalopathy")),
... )
>>>
>>> variations = {}
>>> for i in range(len(data)):
