Using regular expressions (the re module):

>>> import re
>>> NAMES_AND_VALUES = re.compile(r'\w+')
>>> OPERATORS = re.compile(r'(?:\+|\*|\-|\/)+')

>>> string = '((a * 5) // ((10 - y) + z))'

>>> NAMES_AND_VALUES.findall(string)
['a', '5', '10', 'y', 'z']

>>> OPERATORS.findall(string)
['*', '//', '-', '+']
