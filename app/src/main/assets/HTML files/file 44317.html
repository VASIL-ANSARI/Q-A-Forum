       function addMovie(title, rating) {
            var movie = {};

                title = prompt("Enter movie: ");

                rating = parseInt(prompt("Enter rating 1-5 on movie " + (title)));

            movie.ratings = rating; movies.push(movie);
        }
        function printMovies() {
