mylist = [['start', 'a', 'next'],['next','b','end'],['previous','c','start']]
dic={}
for i in mylist:
    dic[i[0]]=i[1:]
print dic

If you dont want list but tuple as value of dict use 

mylist = [['start', 'a', 'next'],['next','b','end'],['previous','c','start']]
dic={}
for i in mylist:
    dic[i[0]]=tuple(i[1:])
print dic
