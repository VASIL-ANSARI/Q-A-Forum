You can use "createViewModel" handler in the component registration, which receives componentInfo as a second argument. "componentInfo.element" is a component's DOMElement. So you can access it and check it's visibility.

ko.components.register('detail', {
    viewModel: {
        createViewModel: function(params, componentInfo) {
            var domElement = componentInfo.element;
            // You can pass domElement as a parameter for the DetailViewModel
            return new DetailViewModel(params);
        }
    },
    template: {require: "text!components/detail.html"}
});
