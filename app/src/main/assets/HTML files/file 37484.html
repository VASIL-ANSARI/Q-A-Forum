Write a simple recursive function:

import numpy as np

data = np.random.randint(0,10,size=4).reshape((1,1,1,1,2,2))

def recursiveIter(d, level=0):
    print level
    if level <= 2:
        for item in d:
            recursiveIter(item, level+1)
    else:
        print d

recursiveIter(data)
