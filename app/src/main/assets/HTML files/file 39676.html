You have a logic error in push. Your logic makes sure that it pushes the input at the bottom of the stack.

What you need is:

void push(StackNode *& top, int num)
{
   StackNode* temp = new StackNode;
   temp->value = num;
   temp->next = top;
   top = temp;
}

pop can also be simplified to:

void pop(StackNode *& top, int &item)
{
   if (top)
   {
      item = top->value;
      StackNode* temp = top;
      top = top->next;
      delete temp; // If you don't delete temp, it is a memory leak.
   }
}

isEmpty can be simplified to:

bool isEmpty(StackNode * top)
{
   return (top == NULL);
}
