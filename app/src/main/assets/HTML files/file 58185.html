Should be:

function initMap() {
  var center = {lat: 37.423021, lng: -122.083739};

  var map = new google.maps.Map(document.getElementById('gmap_holder'), {
    zoom: 4,
    center: center
  });

  var marker = new google.maps.Marker({
    position: center,
    map: map
  });

  // you can set other options
}
