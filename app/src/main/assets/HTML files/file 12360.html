@Override 
public void onBackPressed() { 
    Toast.makeText(this, "onBackPressed", Toast.LENGTH_SHORT).show();
      super.onBackPressed();
      isOnBackeyPressed = true;
      finish();    
}    


@Override
protected void onPause() {
    Toast.makeText(this, "onPause", Toast.LENGTH_SHORT).show();
    super.onPause();
    if (!isOnBackKeyPressed)
       saveState();

}
