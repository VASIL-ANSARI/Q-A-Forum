If you're already using jQuery, you might as well take advantage of its $.ajax function.

$('.radio-msg-type').on('click', function(){
  if ($('#fullQuotation').is(':checked')) {
    $.ajax({
      url: pdf_file,
      type: 'HEAD'
    }).done(function(){
      alert('found');
    }).fail(function(){
      alert('not found');
    });
  }
});
