On init you should find a value of the city form right side text ng-init="rawSelect = (vm.cities| filter: {Text: city.Text})[0]" then Also do change track by for selecting option on init it should be track by po.Text

Markup

<tbody>
    <tr data-ng-repeat="city in vm.cities">
      <td>{{city.Text}}</td>
      <td ng-init="rawSelect = (vm.cities| filter: {Text: city.Text})[0]" >
        <select data-ng-model="rawSelect" data-ng-options="po.Text for po in vm.selectCity track by po.Text">
          <option value="">Match later</option>
        </select>
        <pre>{{rawSelect}}</pre>
      </td>
    </tr>
  </tbody>
