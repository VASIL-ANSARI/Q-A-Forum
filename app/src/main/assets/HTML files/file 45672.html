Your Model class only has one _value property that gets overridden once you set age.

You can instead store your property/value pairs in an object like such:

function Model() {
  this._storage = {};
}

Model.prototype.get = function(property) {
  return this._storage[property];
}

Model.prototype.set = function(property, value) {
  this._storage[property] = value;
}
