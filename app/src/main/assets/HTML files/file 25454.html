You can maintain a counter variable and increment it in your completion callback. When it reaches arr.length, all the files will have been processed.

Something like:

function test()
{
    var processedFiles = 0, fileCount = arr.length;
    for (var i = 0; i < fileCount; i++) {
        saveFileService.SaveFile(arr[i], function() {
            // One file save ended successfully.
            if (++processedFiles == fileCount) {
                // All files have been processed.
            }
        });
    }
}
