After too much of googling, I found the following code which helps me to get the private key from .p12 file using Org.BouncyCastle.Crypto

//This is the function call
 AsymmetricKeyParameter privateKey = TransformRSAPrivateKey(Certi.PrivateKey);

//This is the definition of the function which return Privatekey 

 RSACryptoServiceProvider prov = privateKey as RSACryptoServiceProvider;
        RSAParameters parameters = prov.ExportParameters(true);

        return new RsaPrivateCrtKeyParameters(
            new BigInteger(1, parameters.Modulus),
            new BigInteger(1, parameters.Exponent),
            new BigInteger(1, parameters.D),
            new BigInteger(1, parameters.P),
            new BigInteger(1, parameters.Q),
            new BigInteger(1, parameters.DP),
            new BigInteger(1, parameters.DQ),
            new BigInteger(1, parameters.InverseQ));
