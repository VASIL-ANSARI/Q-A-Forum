create or replace function myfunction(i_company_name varchar2, i_id pls_integer)
return number
as
   l_maxvalue number;
begin
  SELECT Max(some_other_value) 
  into   l_maxvalue
  FROM   table1 
         inner join table2 
                    ON table1.id  = table2.id
                       AND company_name  = i_company_name
  WHERE ID = i_id;
  return l_maxvalue;
end myfunction;
