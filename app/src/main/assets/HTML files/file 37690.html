As according to your comment, you want next 7 days from selected date in dd/mm/yy format.The javascript is

$("#startdate").datepicker({
    dateFormat: 'dd/mm/yy'
});


$("#getDate").click(function () {
    var start = $("#startdate").datepicker("getDate"),
    currentDate = new Date(start),
    between = [];
    for(var i=0;i<7;i++) {
        currentDate.setTime(currentDate.getTime() + 24*60*60*1000);
                                        //miliseconds in 1 day
        between.push(new Date(currentDate));

    }
     for (var i = 0; i < between.length; i++) {
        var date = $.datepicker.formatDate('dd/mm/yy', new Date(between[i]));
        between[i] = date;
    }

    console.log(between)//between is the required result
})
