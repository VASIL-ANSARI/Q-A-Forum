1. keep a reference to the marker rather than creating a new one on the click
  2. change the position of the marker when the click event happens

function initialize() {

  var map = new google.maps.Map(

    document.getElementById("map_canvas"), {

      center: new google.maps.LatLng(30, 30),

      zoom: 13,

      mapTypeId: google.maps.MapTypeId.ROADMAP

    });

  var marker = new google.maps.Marker({

    position: new google.maps.LatLng(30, 30),

    map: map

  });

  $('#btn').on('click', function() {

    marker.setPosition(new google.maps.LatLng(60, 60));

    map.setCenter(marker.getPosition());

  });

  $('#btn2').on('click', function() {

    marker.setPosition(new google.maps.LatLng(30, 30));

    map.setCenter(marker.getPosition());

  });

}

google.maps.event.addDomListener(window, "load", initialize);
