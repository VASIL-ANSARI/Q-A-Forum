datepicker adds a marker class "hasDatepicker" to the element. You have to remove it from the clone:

Updated Fiddle

$(function() {
    $('input.date').datepicker();

    $('button').on('click', function() {
        var x = $('#orig').clone()
        .attr('id', 'else')
        .removeClass("hasDatepicker")    // <=== New line
        .appendTo('body')
        .datepicker();
    });
});
