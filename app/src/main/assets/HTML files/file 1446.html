#!/usr/bin/ruby1.8

class Symbol

  alias_method :old_to_s, :to_s
  def to_s(*args)
    if args == [:upcase]
      old_to_s.upcase
    else
      old_to_s(*args)
    end
  end

end

puts :foo                   # => foo
puts :foo.to_s(:upcase)     # => FOO
