If you set the data-origional attribute on the images to their source then you can run somthing like this on document.ready:

$(function () {
    function load_image($image) {
        if (count >= $images.length) {clearInterval(timer); return false;}
        $image.attr('src', $image.attr('data-origional'));
        count++;
    }
    var $images = $('img'),
        count   = 0,
        timer   = setInterval(function () {
            load_image($images.eq(count));
        }, 800);
});
