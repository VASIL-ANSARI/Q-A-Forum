You can write a recursive generator that yields [value, path] pairs for all scalars in the tree:

function *expand(obj, path=[]) {

    for (let key of Object.keys(obj)) {

        let val = obj[key],

            p = path.concat(key);

        if (typeof val == 'object')

            yield *expand(val, p);

        else

            yield [val, p];

    }

}
