There are many ways to do this.

You could split and than loop over the even indexes and wrap and join back together.

Or you could do all types of regular expression patterns to pull out the pattern

var string = "Argentina,2,Australia,17,Azerbaijan,1,Bangladesh,1,Belgium,2"
var updatedStr = string.replace(/(\D+),(\d+),/g,'"$1",$2,');

But if you want an array at the end, wrapping is the wrong approach. Split the array at the comma and than convert the string to a number in the odd indexes. 

var string = "Argentina,2,Australia,17,Azerbaijan,1,Bangladesh,1,Belgium,2";
var parts = string.split(",");
for (var i=1; i<parts.length; i+=2) {
    parts[i] = parseInt(parts[i],10);
}

And to do the rest of your job for you....

var string = "Argentina,2,Australia,17,Azerbaijan,1,Bangladesh,1,Belgium,2";
var parts = string.split(",");
var out = [];
for (var i=1; i<parts.length; i+=2) {
    parts[i] = parseInt(parts[i],10);
    out.push([parts[i-1],parts[i]]);
}
