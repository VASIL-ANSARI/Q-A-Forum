You could look into remapping the method being passed instead of using routes:

function index(){
    // my default method
}
function _remap($method){
    if ($method == 'en'){
        $this->english();
    } else if($method == 'de') {
        $this->german();
    } else {
        $this->index();
    }
}

function english(){
    // my english method
}

function german(){
    // my german method
}
