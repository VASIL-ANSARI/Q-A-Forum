There is no such method or rule called regex within the jQuery Validate plugin. You'll have to create this custom method using the built-in .addMethod() method.

jQuery.validator.addMethod("regex", function (value, element, param) {
    var regex = new RegExp(param);
    return regex.test(value);
}, "");


$('#DateRangeCriteriaWeek').rules('add', {
    required: true,
    regex: "^(201[3-9]|20[2-9][0-9])-(0[1-9]|[1-4][0-9]|5[0-3])",
    messages: {
        required: "Date Range Criteria is required.",
        regex: "Date Range Criteria format is invalid."
    }
});

See documentation for addMethod: http://jqueryvalidation.org/jQuery.validator.addMethod/
