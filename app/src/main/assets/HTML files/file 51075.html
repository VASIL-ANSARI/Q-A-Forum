$(document).ready(function(){
    $('#myselect').change(function(){
            if ($(this).val() === "Student") {
                    $('#staffForm').hide();
                    $('#studentForm').show();
            } else {
                    $('#studentForm').hide();
                    $('#staffForm').show();
            }
    });
});
