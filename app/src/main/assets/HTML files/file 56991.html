In your ajax loop through the data and append to the view:

success: function(data) {
    var dataView = $('#dataTable tbody');
    dataView.empty();
    for (dataRow in data.datos) {
        dataView.append('<tr><td>' 
                        + dataRow.variable1
                        + '</td><td>'
                        + dataRow.variable2
                        + '</td><td>'
                        + dataRow.variable3
                        + '</td><td>'
                        + dataRow.variable4
                        + '</td><td>'
                        + dataRow.variable5
                        + '</td><td>'
                        + dataRow.variable6
                        + '</td></tr>');
    }
},

Replace variable1 - variable6 names for your database field names
