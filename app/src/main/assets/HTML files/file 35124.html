Not really. You can also create an IIF for obj though:

obj = function () {
    var privatething = 'hithere';
    return {
        get value() {
            return privatething;
        },
        set value(v) {
            privatething = v;
        }
    };
}();
obj.value; //=> 'hithere';
obj.value = 'good morning to you too';
obj.value; //=> 'good morning to you too'
